# CELL 3 — SECTION 1: Setup Widgets  Edit Table & Column Descriptions

# HOW TO USE:
#   1. Pick a table from Widget 1
#   2. See the current descriptions printed below
#   3. Pick a column from Widget 3
#   4. Fill in Widget 2 (table desc) and Widget 4 (column desc)
#   5. Run Cell 4 to stage the changes

dbutils.widgets.removeAll()

# Widget 1 — Table selector
dbutils.widgets.dropdown(
    name         = "1_table",
    defaultValue = table_ids()[0] if table_ids() else "",
    choices      = table_ids(),
    label        = "1. Select Table"
)

selected_table = dbutils.widgets.get("1_table")
t = find_table(selected_table)

print("=" * 60)
print("  SECTION 1 — Edit Table & Column Descriptions")
print("=" * 60)

if not t:
    print("No table selected or table not found.")
else:
    existing_table_desc = (t.get("description") or [""])[0]
    col_configs         = t.get("column_configs", [])
    col_names           = [c["column_name"] for c in col_configs]

    print(f"\n  Selected Table : {selected_table}")
    print(f"\n  Current Table Description:")
    print(f"      '{existing_table_desc or '(empty)'}'")
    print(f"\n  Configured Columns ({len(col_names)}):")
    if col_configs:
        for c in col_configs:
            cdesc = c.get("description", "(no description)")
            print(f"      • {c['column_name']:<30} {cdesc[:60]}")
    else:
        print("      (none — add columns when using Section 2 Add Table)")

    # Widget 2 — New table description (pre-filled with existing)
    dbutils.widgets.text(
        name         = "2_table_description",
        defaultValue = existing_table_desc,
        label        = "2. New Table Description (edit me)"
    )

    # Widget 3 — Column selector
    col_choices = col_names if col_names else ["(no columns configured)"]
    dbutils.widgets.dropdown(
        name         = "3_column",
        defaultValue = col_choices[0],
        choices      = col_choices,
        label        = "3. Select Column to Edit"
    )

    selected_col      = dbutils.widgets.get("3_column")
    col_obj           = next((c for c in col_configs if c["column_name"] == selected_col), {})
    existing_col_desc = col_obj.get("description", "")

    # Widget 4 — New column description (pre-filled with existing)
    dbutils.widgets.text(
        name         = "4_column_description",
        defaultValue = existing_col_desc,
        label        = "4. New Column Description (edit me)"
    )

    print(f"\n  Selected Column     : {selected_col}")
    print(f"  Current Column Desc : '{existing_col_desc or '(empty)'}'")
    print("\n Edit widgets 2 and 4 above, then run Cell 4 to stage changes.")
