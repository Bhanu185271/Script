def getVersionNumber(deltaPath: str) -> int:
  """
  A function to get the max Version number on the c b 
  
  Warning:
  --------
  Including '/dbfs/' in the file path will throw error.
  
  Inputs:
  -------
  deltaPath: type == str
              Provide the full path of the delta location for which you want to check the max version.
              
  Returns:
  --------
  versionNumberForCurrentLoad: as an int value
                                This is the version number that will be added to the current data load
  """
  try:
    sdf = spark.read.format("delta").load(deltaPath)
  except:
    # An exception will be raised if we dont have any data ie. there is no delta table existing ie on our first load
    print("Something bad happened")
    print("|--> We have no delta table in all likelihood...")
    versionNumberForCurrentLoad = 1
    print(f">> Setting this as first version with {versionNumberForCurrentLoad}...")
  else:
    maxVersion = (spark.read
                         .format("delta")
                         .load(deltaPath)
                         .select('Version')
                         .distinct()
                         .agg(max('Version').alias('MaxVersion'))
                  ).collect()[0]['MaxVersion']
    versionNumberForCurrentLoad = maxVersion + 1
    print(f"|--> Max Version in Existing Data : {maxVersion}\nVersion For Current Load : {versionNumberForCurrentLoad}")
  finally:
    return versionNumberForCurrentLoad

def firstLoad(partitionList,cleanedDF,enrichedLayerPath): 
        (cleanedDF.write
               .format("delta")
               .mode("overwrite")
               .partitionBy(partitionList)
               .save(enrichedLayerPath)
        )

def incrementalLoadOverwrite(partitionList,mainDF,enrichedLayerPath): 
# Overwriting main dataset
        (mainDF.write
               .format("delta")
               .mode("overwrite")
               .partitionBy(partitionList)
               .save(enrichedLayerPath)
        )
    
def incrementalLoadAppend(partitionList,cleanedDF,enrichedLayerPath): 
# Appending latest data to enrich layer
        (cleanedDF.write
               .format("delta")
               .mode("append")
               .partitionBy(partitionList)
               .save(enrichedLayerPath)
        )

def renameColumns(cleanedDF, columns):
    for old_name, new_name in columns.items():
        cleanedDF = cleanedDF.withColumnRenamed(old_name, new_name)
    return cleanedDF
  
def withColumnsQuarterAndYear(cleanedDF, columnsDict):
    for old_column, derived_column in columnsDict.items():
       for i in range(len(derived_column) - 1):
        cleanedDF = cleanedDF.withColumn(derived_column[i], quarter(col(old_column)))
        cleanedDF = cleanedDF.withColumn(derived_column[i+1], year(col(old_column)))
    return cleanedDF

def removeSpacesInColumnList(pharma_xl_df):
  for i in pharma_xl_df.columns:
    if pharma_xl_df[i].dtype==object:
      pharma_xl_df[i] = pharma_xl_df[i].astype(str).str.strip()
    rename = i.lower().strip().replace(" ",'_') #rename = i.lower().strip().replace(" ",'_')
    pharma_xl_df.rename(columns={i:rename},inplace=True)
    #print(rename)
  return pharma_xl_df

def writeToEnriched(country,foundationLayerPath,enrichedLayerPath,dataset,channel,partitionList,requiredColumnsForEnrichedLayer,data,columnsDict,incrementalColumnValue):
  
  """
   Utility function writes data from foundation to enrich layer delta path. 
   
  Inputs:
  ------
  country:
        The country for which we are running this merge operation
  foundationLayerPath: 
        The delta path in foundation layer from where data is read. 
              
  enrichedLayerPath:
        The delta path in enriched layer from where data is written. 
                        
  dataset:
        Sourcefile which we are running for.
  
  channel:
        Indirect/Independent
            
  partitionList: 
        The list based on which partition is done in delta path of enriched layer.
                 
  requiredColumnsForEnrichedLayer:
        The list of generic column names to be ingested to enriched layer.
                              
  data:
        Dictionary containing columns to be renamed as per generic schema.
  
  columnForMergeLogic:
        This is the column based on which merge operation is performed in enriched layer.
  
  columnsQuarter:
        Dictionary containing (quarter) column to exracted from date column.
  
  columnsYear:
        Dictionary containing (year) column to exracted from date column.
      
  """
   
  """
  Reading latest version data from foundation layer
  """ 
  cnt=0
    
  try:
    foundationDF=spark.read.format("delta").load(foundationLayerPath)
    enrichDF=spark.read.format("delta").load(enrichedLayerPath)
    print("validating foundation and enrich")
    MaxVersionInFoundationLayer=foundationDF.agg({'Version': 'Max'})
    MaxVersionDistinctFoundation=MaxVersionInFoundationLayer.collect()[0]['max(Version)']
    MaxVersionDistinctFoundation
    MaxVersionInEnrichLayer=enrichDF.agg({'Version': 'Max'})
    MaxVersionDistinctEnriched=MaxVersionInEnrichLayer.collect()[0]['max(Version)']
    MaxVersionDistinctEnriched
    if MaxVersionDistinctFoundation <= MaxVersionDistinctEnriched:
      print("Foundation is not greater than or same as enrich version")
      message="Enrich layer is already uptodate"
      cnt=0
      return_res=json.dumps({"Status": 0,"Counts": cnt,"Result": message,"code":"NA"})
      return return_res
    MaxVersion=foundationDF.agg({'Version': 'max'})
    LatestVersion=str(MaxVersion.collect()[0]['max(Version)'])
    unCleanedDF=foundationDF.filter(foundationDF.Version == LatestVersion) 

    print("Cleaning Started\n")
    cleanedDF = unCleanedDF

        res=FcstOutputFullMergeLogic(country,cleanedDF,enrichDF,columnsDict,partitionList,enrichedLayerPath)

      elif dataset == "FcstOitItem":

      elif dataset == "Pharmacy":
