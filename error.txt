def fix_description(val):
    if isinstance(val, list):
        clean = [str(v) for v in val if v is not None and str(v).strip()]
        return clean if clean else None
    if isinstance(val, str) and val.strip():
        return [val.strip()]
    return None

def sanitize_descriptions(ss):
    for t in ss.get("data_sources", {}).get("tables", []):
        if not isinstance(t, dict): continue
        if "description" in t:
            fixed = fix_description(t["description"])
            if fixed: t["description"] = fixed
            else:     del t["description"]
        for c in t.get("column_configs", []):
            if not isinstance(c, dict): continue
            if "description" in c:
                fixed = fix_description(c["description"])
                if fixed: c["description"] = fixed
                else:     del c["description"]

def collect_space1_ids(ss1_original):
    ids = set()
    def walk(obj):
        if isinstance(obj, dict):
            if "id" in obj and isinstance(obj["id"], str):
                ids.add(obj["id"])
            for v in obj.values():
                walk(v)
        elif isinstance(obj, list):
            for item in obj:
                walk(item)
    walk(ss1_original)
    return ids

def strip_space2_ids(ss, space1_ids):
    def walk(obj):
        if isinstance(obj, dict):
            if "id" in obj and isinstance(obj["id"], str):
                if obj["id"] not in space1_ids:
                    del obj["id"]
            for v in list(obj.values()):
                walk(v)
        elif isinstance(obj, list):
            for item in obj:
                walk(item)
    walk(ss)

def sort_lists(ss):
    """
    Sort all lists as the API requires.
    Items without an id (stripped Space 2 items) sort to the END
    so Space 1 items (with ids) always come first.
    """
    def by_id(lst):
        return sorted(
            lst,
            key=lambda x: (
                0 if (isinstance(x, dict) and x.get("id"))  else 1,  # no-id → end
                x.get("id", "") if isinstance(x, dict) else ""
            )
        )

    def by_identifier(lst):
        return sorted(
            lst,
            key=lambda x: x.get("identifier", "") if isinstance(x, dict) else ""
        )

    ds   = ss.get("data_sources", {})
    inst = ss.get("instructions", {})
    snip = inst.get("sql_snippets", {})

    # Tables sorted by identifier (no id field on tables)
    if "tables" in ds:
        ds["tables"] = by_identifier(ds["tables"])

    # All instruction lists sorted by id
    for k in ("text_instructions", "example_question_sqls"):
        if k in inst:
            inst[k] = by_id(inst[k])

    for k in ("filters", "expressions", "measures"):
        if k in snip:
            snip[k] = by_id(snip[k])

    # Benchmarks
    bm = ss.get("benchmarks", {})
    if isinstance(bm, dict) and "questions" in bm:
        bm["questions"] = by_id(bm["questions"])

    # Sample questions
    cfg = ss.get("config", {})
    if "sample_questions" in cfg:
        cfg["sample_questions"] = by_id(cfg["sample_questions"])

```
## What Changed

The old order was:
```
1. Sanitize descriptions
2. Sort by id          ← Space 2 IDs still present, sort works
3. Strip Space 2 IDs   ← IDs removed AFTER sort → order now broken
```

The new order is:
```
1. Sanitize descriptions
2. Strip Space 2 IDs   ← remove first
3. Sort by id          ← sort after, items without id go to END stably
