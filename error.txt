    def norm_key(s):
        """
          normalisation so visually identical strings always match:
          1. NFKC unicode normalisation
          2. Collapse ALL whitespace (spaces, tabs, newlines) to single space
          3. Strip leading/trailing whitespace
          4. Casefold 
        """
        s = unicodedata.normalize("NFKC", str(s))
        s = re.sub(r"\s+", " ", s)
        return s.strip().casefold()

    def make_desc(cat, name, desc, sql, addl):
        p = []
        if cat == "INSTRUCTION":
            return name.strip() or "(empty)"

        elif cat == "TABLE":
            # Table name always first
            p.append(f"Table  : {name}")
            if desc:
                p.append(f"Desc   : {desc}")
            if addl:
                p.append(addl)
            return "\n".join(p)

        elif cat == "JOIN":
            if sql:  p.append(f"SQL    : {sql}")
            return "\n".join(p) if p else name

        elif cat in ("MEASURE", "EXPRESSION", "FILTER"):
            if desc: p.append(f"Desc   : {desc}")
            if sql:  p.append(f"SQL    : {sql}")
            if addl: p.append(addl)
            return "\n".join(p) if p else name

            p.append(f"Q      : {name}")

            return "\n".join(p)

            return desc or name

    # Categories where SQL content is meaningful for secondary matching
    SQL_CATS = {"JOIN", "MEASURE", "EXPRESSION", "FILTER", "BENCHMARK", "EXAMPLE_SQL"}

    #  Status constants 
    IDENTICAL = "IDENTICAL"
    MODIFIED  = "MODIFIED"
    SP1_ONLY  = "SP1_ONLY"
    SP2_ONLY  = "SP2_ONLY"

    # Background colours per status 
    STATUS_BG = {
        IDENTICAL : "",              # no background = white
        MODIFIED  : "on #fff3cd",    # amber / yellow
        SP1_ONLY  : "on #d4edda",    # light green
        SP2_ONLY  : "on #cce5ff",    # light blue

    # status labels 
    STATUS_LABEL = {
        IDENTICAL : "BOTH",
        MODIFIED  : "MODIFIED",
        SP1_ONLY  : "<- SP1 ONLY",
        SP2_ONLY  : "-> SP2 ONLY",
    }

    # Sort order: MODIFIED first, then SP1/SP2-only, IDENTICAL last
    STATUS_ORDER = {MODIFIED: 0, SP1_ONLY: 1, SP2_ONLY: 2, IDENTICAL: 3}

    #  SQL normaliser
    def norm_sql(val):
        """
        Normalise SQL for comparison:
          1. NFKC unicode
          2. Collapse ALL whitespace (spaces, tabs, newlines) → single space
          3. Strip
          4. Casefold
        Returns "" for None / empty.
        """
        if val is None:
            return ""
        if isinstance(val, list):
            val = "".join(str(x) for x in val)
        s = unicodedata.normalize("NFKC", str(val))
        s = re.sub(r"\s+", " ", s)
        return s.strip().casefold()

    #  Content-level diff detector 
    def content_status(r1, r2):
        """
        Given two rows that have been matched (both exist),
        determine whether they are IDENTICAL or MODIFIED.
        Returns (status_str, list_of_changed_field_names).
        """
        sql_match  = norm_sql(r1["sql"])         == norm_sql(r2["sql"])
        desc_match = norm_key(r1["description"]) == norm_key(r2["description"])
        name_match = norm_key(r1["name"])        == norm_key(r2["name"])

        if sql_match and desc_match and name_match:
            return IDENTICAL, []

        diffs = []
        if not name_match: diffs.append("name")
        if not sql_match:  diffs.append("SQL")
        if not desc_match: diffs.append("description")
        return MODIFIED, diffs

    #  Two pass matcher 
    def two_pass_match(sp1_df, sp2_df, cat):
        """
        Match rows from sp1_df against sp2_df using two passes:
          Pass 1: by norm_key (name/question)
          Pass 2: by norm_sql (SQL content) — only for SQL_CATS

        Handles duplicates: if multiple rows share the same norm_key in one
        space, each is treated as a separate entry (first matches, rest become
        SP1/SP2-only with a DUP flag).

        Returns a list of match dicts:
          {
            "r1"         : row or None,
            "r2"         : row or None,
            "status"     : IDENTICAL | MODIFIED | SP1_ONLY | SP2_ONLY,
            "diffs"      : [...],
            "match_type" : "name" | "sql" | None,
            "dup_flag"   : True if this is a duplicate entry,
          }
        """
        results = []

        # Collect ALL rows (allow duplicates) — list of norm_key, sql_key, row
        def collect(df):
            items = []
            for _, r in df.iterrows():
                nk = r["norm_key"]
                sk = norm_sql(r["sql"]) if cat in SQL_CATS else ""
                items.append({"nk": nk, "sk": sk, "row": r, "matched": False})
            return items

        sp1_items = collect(sp1_df)
        sp2_items = collect(sp2_df)

        #  Pass 1: match by norm_key 
        for i1, e1 in enumerate(sp1_items):
            for i2, e2 in enumerate(sp2_items):
                if e1["matched"] or e2["matched"]:
                    continue
                if e1["nk"] == e2["nk"]:
                    status, diffs = content_status(e1["row"], e2["row"])
                    results.append({
                        "r1": e1["row"], "r2": e2["row"],
                        "status": status, "diffs": diffs,
                        "match_type": "name", "dup_flag": False,
                    })
                    e1["matched"] = True
                    e2["matched"] = True
                    break  # one-to-one matching

        #  Pass 2: match unmatched by SQL content
        if cat in SQL_CATS:
            unmatched_sp1 = [e for e in sp1_items if not e["matched"]]
            unmatched_sp2 = [e for e in sp2_items if not e["matched"]]

            for e1 in unmatched_sp1:
                if not e1["sk"]:          # no SQL : nothing to match on
                    continue
                for e2 in unmatched_sp2:
                    if e2["matched"] or not e2["sk"]:
                        continue
                    if e1["sk"] == e2["sk"]:
                        status, diffs = content_status(e1["row"], e2["row"])
                        # SQL matched : if name/desc differ that counts as MODIFIED
                        results.append({
                            "r1": e1["row"], "r2": e2["row"],
                            "status": status, "diffs": diffs,
                            "match_type": "sql", "dup_flag": False,
                        })
                        e1["matched"] = True
                        e2["matched"] = True
                        break

        #  Remaining unmatched → SP1_ONLY / SP2_ONLY 
        for e1 in sp1_items:
            if not e1["matched"]:
                results.append({
                    "r1": e1["row"], "r2": None,
                    "status": SP1_ONLY, "diffs": [],
                    "match_type": None, "dup_flag": False,
                })
        for e2 in sp2_items:
            if not e2["matched"]:
                results.append({
                    "r1": None, "r2": e2["row"],
                    "status": SP2_ONLY, "diffs": [],
                    "match_type": None, "dup_flag": False,
                })

        # Sort: MODIFIED first, then SP1/SP2-only, IDENTICAL last;
        #       within same status, alphabetical by name
        def sort_key(m):
            r = m["r1"] if m["r1"] is not None else m["r2"]
            return (STATUS_ORDER[m["status"]], norm_key(r["name"]))

        results.sort(key=sort_key)
        return results

    # Per category rich comparison tables 
    #
    # One Table per category
    # 5 columns: Status | SP1 Serial | SP1 Description | SP2 Serial | SP2 Description
 
    def make_compare_table(cat_matches, sp1_title, sp2_title):
        tbl = Table(
            box          = box.SIMPLE_HEAVY,
            show_lines   = True,
            expand       = True,
            padding      = (0, 1),
            header_style = "bold white on #37474f",

                       style="bold #003399", width=10, no_wrap=True)
        tbl.add_column("SP2 Description / SQL / Details",
                       style="#000000", overflow="fold", ratio=1)

        for m in cat_matches:
            status = m["status"]

            tbl.add_row(
                label,
                m["SP1_serial"],
                m["SP1_description"],
                m["SP2_serial"],
                m["SP2_description"],
                style=rs,
            )
        return tbl

    def fix_desc(v):
        if isinstance(v, list):
            c = [str(x) for x in v if x is not None and str(x).strip()]
            return c if c else None
        if isinstance(v, str) and v.strip(): return [v.strip()]
        return None

    def sanitize(ss):
        for t in ss.get("data_sources", {}).get("tables", []):
            if not isinstance(t, dict): continue
            if "description" in t:
                f = fix_desc(t["description"])
                if f: t["description"] = f
                else: del t["description"]
            for c in t.get("column_configs", []):
                if not isinstance(c, dict): continue
                if "description" in c:
                    f = fix_desc(c["description"])
                    if f: c["description"] = f
                    else: del c["description"]
                # c["hidden"] NOT touched — preserved exactly

    def sort_all(ss):
        by_id    = lambda l: sorted(l, key=lambda x: x.get("id","zzz") if isinstance(x,dict) else "zzz")
        by_ident = lambda l: sorted(l, key=lambda x: x.get("identifier","") if isinstance(x,dict) else "")
        ds, inst = ss.get("data_sources", {}), ss.get("instructions", {})
        snip = inst.get("sql_snippets", {})
        bm   = ss.get("benchmarks", {})
        cfg  = ss.get("config", {})
        if "tables"                in ds:   ds["tables"]                  = by_ident(ds["tables"])
        if "metric_views"          in ds:   ds["metric_views"]            = by_ident(ds["metric_views"])
        if "example_question_sqls" in inst: inst["example_question_sqls"] = by_id(inst["example_question_sqls"])
        for k in ("filters", "expressions", "measures"):
            if k in snip: snip[k] = by_id(snip[k])
        if isinstance(bm, dict) and "questions" in bm: bm["questions"] = by_id(bm["questions"])

    def write_json(path, data, label):
        try:
            os.makedirs(os.path.dirname(path), exist_ok=True)
            with open(path, "w", encoding="utf-8") as f:
                json.dump(data, f, indent=2, ensure_ascii=False)
            console.print(f"  ✓  {label} → {path}")
        except OSError as e:
            console.print(f"  ⚠  {label}: workspace write failed ({e}) — using DBFS")
            fb = f"{DBFS_SNAPSHOT}/{os.path.basename(path)}"
            dbutils.fs.put(fb, json.dumps(data, indent=2, ensure_ascii=False), overwrite=True)
            console.print(f"  ✓  {label} → {fb}  (DBFS)")

    def copy_file(src, dst, label):
        try:
            os.makedirs(os.path.dirname(dst), exist_ok=True)
            shutil.copy2(src, dst)
            console.print(f"  ✓  {label} → {dst}")
        except OSError as e:
            console.print(f"  ⚠  {label}: workspace copy failed ({e}) — using DBFS")
            fb = f"{DBFS_SNAPSHOT}/{os.path.basename(dst)}"
