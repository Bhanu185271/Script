    # Human-readable labels for each category
    CAT_LABELS = {
        "INSTRUCTION"    : "Genie Instructions",
        "TABLE"          : "Table Details",
        "JOIN"           : "Join Details",
        "MEASURE"        : "Measures",
        "EXPRESSION"     : "Expressions",
        "FILTER"         : "Filters",
        "BENCHMARK"      : "Benchmarks",
        "SAMPLE_QUESTION": "Sample Questions",
        "EXAMPLE_SQL"    : "Example SQLs",

    # Fetch a Genie space and return the outer metadata and parsed serialized space
    def fetch_space(sid):
        r = requests.get(
            f"{DATABRICKS_HOST}/api/2.0/genie/spaces/{sid}",
            headers=HEADERS,
            params={"include_serialized_space": "true"},
            timeout=30,
        )
        if r.status_code != 200:
            raise Exception(f"HTTP {r.status_code} for '{sid}': {r.text[:300]}")
        outer = r.json()
        raw   = outer.get("serialized_space")
        if raw is None:
            raise KeyError(f"'serialized_space' missing. Keys: {list(outer.keys())}")
        return outer, (json.loads(raw) if isinstance(raw, str) else raw)

    # Helper: safely convert any value to a string
    def safe_str(v):
        if v is None:           return ""
        if isinstance(v, list): return " | ".join(str(x) for x in v if x is not None)
        return str(v)

    # Helper: convert SQL field (string or list) to a plain string
    def sql_str(v):
        if isinstance(v, list): return "".join(str(x) for x in v)
        if isinstance(v, str):  return v
        return ""

    # Helper: normalize a key for comparison (unicode, whitespace, case)
    def norm_key(s):
        s = unicodedata.normalize("NFKC", str(s))
        s = re.sub(r"\s+", " ", s)
        return s.strip().casefold()

    # Section accessors for each part of the serialized space
    def get_tables(ss):    return ss.get("data_sources", {}).get("tables", [])
    def get_inst(ss):      return ss.get("instructions", {})
    def get_snip(ss):      return get_inst(ss).get("sql_snippets", {})
    def get_txt(ss):       return get_inst(ss).get("text_instructions", [])
    def get_sqls(ss):      return get_inst(ss).get("example_question_sqls", [])
    def get_joins(ss):     return get_inst(ss).get("join_specs", [])
    def get_filters(ss):   return get_snip(ss).get("filters", [])
    def get_exprs(ss):     return get_snip(ss).get("expressions", [])
    def get_measures(ss):  return get_snip(ss).get("measures", [])
    def get_benchmarks(ss):
        bm = ss.get("benchmarks", {})
        return (bm.get("questions", []) if isinstance(bm, dict)
                else bm if isinstance(bm, list) else [])
    def get_sample_qs(ss): return ss.get("config", {}).get("sample_questions", [])

    # Build a display string for each item based on its category
    def make_desc(cat, name, desc, sql, addl):
        p = []
        if cat == "INSTRUCTION":
            return name.strip() or "(empty)"
        elif cat == "TABLE":
            p.append(f"Table  : {name}")
            if desc: p.append(f"Desc   : {desc}")
            if addl: p.append(addl)
            return "\n".join(p)
        elif cat == "JOIN":
            if sql: p.append(f"SQL    : {sql}")
            return "\n".join(p) if p else name
        elif cat in ("MEASURE", "EXPRESSION", "FILTER"):
            if desc: p.append(f"Desc   : {desc}")
            if sql:  p.append(f"SQL    : {sql}")
            if addl: p.append(addl)
            return "\n".join(p) if p else name
        elif cat == "BENCHMARK":
            p.append(f"Q      : {name}")
            if sql: p.append(f"SQL    : {sql}")
            return "\n".join(p)
        elif cat == "SAMPLE_QUESTION":
            return name
        elif cat == "EXAMPLE_SQL":
            p.append(f"Q      : {name}")
            if sql: p.append(f"SQL    : {sql}")
            return "\n".join(p)
        else:
            return desc or name

    # Build a row list and a registry dict for all items in a space
    def build_registry(ss, prefix):
        rows, registry, ctr = [], {}, [1]

        def nxt():
            s = f"{prefix}_{ctr[0]:03d}"; ctr[0] += 1; return s

        def add(cat, key, name, desc, sql, addl, detail, item):
            sn = nxt()
            rows.append({
                "serial_no"   : sn,
                "category"    : cat,
                "name"        : name,
                "description" : desc,
                "sql"         : sql,
                "additional"  : addl,
                "full_detail" : detail,
                "display_desc": make_desc(cat, name, desc, sql, addl),
                "norm_key"    : norm_key(key),
            })
            registry[sn] = {"category": cat, "key": key, "item": copy.deepcopy(item)}

        # 1. Instructions - each line becomes a separate registry entry
        lines = []
        for ti in get_txt(ss):
            if not isinstance(ti, dict): continue
            c = ti.get("content", [])
            if isinstance(c, str): c = [c]
            for ln in c:
                ln = str(ln).strip()
                if ln: lines.append((ln, ti))
        for ln, ti in sorted(lines, key=lambda x: x[0].lower()):
            add("INSTRUCTION", ln, ln, "", "", "",
                json.dumps({"line": ln}), {"line": ln, "_parent_ti": ti})

        # 2. Tables
        for t in sorted([t for t in get_tables(ss)
                         if isinstance(t, dict) and t.get("identifier")],
                        key=lambda t: t["identifier"].lower()):
            ident = t["identifier"]
            cols  = t.get("column_configs", [])
            vis   = [c for c in cols if isinstance(c, dict) and not c.get("hidden", False)]
            hid   = [c for c in cols if isinstance(c, dict) and     c.get("hidden", False)]

            addl_lines = [f"Cols   : {len(cols)} total  (visible:{len(vis)}  hidden:{len(hid)})"]
            if vis:
                addl_lines.append("Visible Columns:")

                    col_name = c.get("column_name", "?")
                    col_desc = safe_str(c.get("description", ""))
                    addl_lines.append(
                        f"  - {col_name}" + (f"  :  {col_desc}" if col_desc else ""))
            if hid:
                addl_lines.append("Hidden Columns:")
                for c in hid:
                    addl_lines.append(f"  - {c.get('column_name', '?')}")

            add("TABLE", ident, ident, safe_str(t.get("description", "")),
                "", "\n".join(addl_lines), json.dumps(t, indent=2), t)

        # 3. Joins
        def jk(j):
            l = j.get("left", {}).get("identifier", "") if isinstance(j.get("left"), dict) else ""
            r = j.get("right", {}).get("identifier", "") if isinstance(j.get("right"), dict) else ""
            return f"{min(l, r)} <-> {max(l, r)}"
        for j in sorted([j for j in get_joins(ss) if isinstance(j, dict)],
                        key=lambda j: jk(j).lower()):
            k = jk(j)
            add("JOIN", k, k, "", sql_str(j.get("sql", "")), "",
                json.dumps(j, indent=2), j)

        # 4. Measures
        for m in sorted([m for m in get_measures(ss) if isinstance(m, dict)],
                        key=lambda m: m.get("display_name", m.get("alias", "")).lower()):
            n    = m.get("display_name", m.get("alias", ""))
            syns = ", ".join(m.get("synonyms", [])) if isinstance(m.get("synonyms"), list) else ""
            add("MEASURE", n, n, safe_str(m.get("description", "")),
                sql_str(m.get("sql", "")), f"Synonyms: {syns}" if syns else "",
                json.dumps(m, indent=2), m)

        # 5. Expressions
        for e in sorted([e for e in get_exprs(ss) if isinstance(e, dict)],
                        key=lambda e: e.get("alias", "").lower()):
            a = e.get("alias", "")
            add("EXPRESSION", a, a, safe_str(e.get("description", "")),
                sql_str(e.get("sql", "")), "", json.dumps(e, indent=2), e)

        # 6. Filters
        for f in sorted([f for f in get_filters(ss) if isinstance(f, dict)],
                        key=lambda f: f.get("display_name", "").lower()):
            n = f.get("display_name", "")

                sql_str(f.get("sql", "")), "", json.dumps(f, indent=2), f)

        # 7. Benchmarks
        for b in sorted([b for b in get_benchmarks(ss) if isinstance(b, dict)],
                        key=lambda b: safe_str(b.get("question", "")).lower()):
            q = safe_str(b.get("question", ""))
            add("BENCHMARK", q, q, "", sql_str(b.get("sql", "")), "",
                json.dumps(b, indent=2), b)

        # 8. Sample Questions
        for sq in sorted([sq for sq in get_sample_qs(ss) if isinstance(sq, dict)],
                         key=lambda sq: safe_str(sq.get("question", "")).lower()):
            q = safe_str(sq.get("question", ""))
            add("SAMPLE_QUESTION", q, q, "", "", "", json.dumps(sq), sq)

        # 9. Example SQLs
        for es in sorted([es for es in get_sqls(ss) if isinstance(es, dict)],
                         key=lambda es: safe_str(es.get("question", "")).lower()):
            q = safe_str(es.get("question", ""))
            add("EXAMPLE_SQL", q, q, "", sql_str(es.get("sql", "")), "",
                json.dumps(es), es)

        return rows, registry

    # Fetch both spaces and build their registries
    outer1, ss1 = fetch_space(SPACE_ID_1)
    outer2, ss2 = fetch_space(SPACE_ID_2)

                label = CAT_LABELS.get(cat, cat)

                tbl.add_section()
            tbl.add_row(row["serial_no"], row["category"], row["display_desc"])

        console.print(); console.print(tbl); console.print()

    render_space_table(

# CELL 3 - SIDE-BY-SIDE COMPARISON

mode = dbutils.widgets.get("mode")

if mode == "merge":

    CATEGORIES = [
        "INSTRUCTION", "TABLE", "JOIN", "MEASURE", "EXPRESSION",
        "FILTER", "BENCHMARK", "SAMPLE_QUESTION", "EXAMPLE_SQL",
    ]

    SQL_CATS = {"JOIN", "MEASURE", "EXPRESSION", "FILTER", "BENCHMARK", "EXAMPLE_SQL"}

    IDENTICAL = "IDENTICAL"
    MODIFIED  = "MODIFIED"
    SP1_ONLY  = "SP1_ONLY"
    SP2_ONLY  = "SP2_ONLY"

    STATUS_BG = {
        IDENTICAL : "",
        MODIFIED  : "on #fff3cd",
        SP1_ONLY  : "on #d4edda",
        SP2_ONLY  : "on #cce5ff",
    }

    STATUS_LABEL = {
        IDENTICAL : "BOTH",
        MODIFIED  : "MODIFIED",
        SP1_ONLY  : "SP1 ONLY",
        SP2_ONLY  : "SP2 ONLY",
    }

    STATUS_ORDER = {MODIFIED: 0, SP1_ONLY: 1, SP2_ONLY: 2, IDENTICAL: 3}

    # Normalize SQL for comparison
    def norm_sql(val):
        if val is None:
            return ""
        if isinstance(val, list):
            val = "".join(str(x) for x in val)
        s = unicodedata.normalize("NFKC", str(val))
        s = re.sub(r"\s+", " ", s)
        return s.strip().casefold()

    # Check whether two matched rows are identical or modified
    def content_status(r1, r2):
        sql_match  = norm_sql(r1["sql"])         == norm_sql(r2["sql"])
        desc_match = norm_key(r1["description"]) == norm_key(r2["description"])
        name_match = norm_key(r1["name"])        == norm_key(r2["name"])
        if sql_match and desc_match and name_match:
            return IDENTICAL, []
        diffs = []
        if not name_match: diffs.append("name")
        if not sql_match:  diffs.append("SQL")
        if not desc_match: diffs.append("description")
        return MODIFIED, diffs

    # Match items from two spaces - first by name, then by SQL for SQL categories
    def two_pass_match(sp1_df, sp2_df, cat):
        results = []

        def collect(df):
            items = []
            for _, r in df.iterrows():
                nk = r["norm_key"]
                sk = norm_sql(r["sql"]) if cat in SQL_CATS else ""
                items.append({"nk": nk, "sk": sk, "row": r, "matched": False})
            return items

        sp1_items = collect(sp1_df)
        sp2_items = collect(sp2_df)

        # Pass 1: match by name
        for i1, e1 in enumerate(sp1_items):
            for i2, e2 in enumerate(sp2_items):
                if e1["matched"] or e2["matched"]:
                    continue
                if e1["nk"] == e2["nk"]:
                    status, diffs = content_status(e1["row"], e2["row"])
                    results.append({
                        "r1": e1["row"], "r2": e2["row"],
                        "status": status, "diffs": diffs,
                        "match_type": "name", "dup_flag": False,
                    })
                    e1["matched"] = True
                    e2["matched"] = True
                    break

        # Pass 2: match remaining SQL-category items by SQL content
        if cat in SQL_CATS:
            unmatched_sp1 = [e for e in sp1_items if not e["matched"]]
            unmatched_sp2 = [e for e in sp2_items if not e["matched"]]
            for e1 in unmatched_sp1:
                if not e1["sk"]: continue
                for e2 in unmatched_sp2:
                    if e2["matched"] or not e2["sk"]: continue
                    if e1["sk"] == e2["sk"]:
                        status, diffs = content_status(e1["row"], e2["row"])
                        results.append({
                            "r1": e1["row"], "r2": e2["row"],
                            "status": status, "diffs": diffs,
                            "match_type": "sql", "dup_flag": False,
                        })
                        e1["matched"] = True
                        e2["matched"] = True
                        break

        # Remaining unmatched items are space-only
        for e1 in sp1_items:
            if not e1["matched"]:
                results.append({
                    "r1": e1["row"], "r2": None,
                    "status": SP1_ONLY, "diffs": [],
                    "match_type": None, "dup_flag": False,
                })
        for e2 in sp2_items:
            if not e2["matched"]:
                results.append({
                    "r1": None, "r2": e2["row"],
                    "status": SP2_ONLY, "diffs": [],
                    "match_type": None, "dup_flag": False,
                })

        def sort_key(m):
            r = m["r1"] if m["r1"] is not None else m["r2"]
            return (STATUS_ORDER[m["status"]], norm_key(r["name"]))

        results.sort(key=sort_key)
        return results

    compare_rows = []
    for cat in CATEGORIES:
        sp1_cat = df_sp1[df_sp1["category"] == cat].copy()
        sp2_cat = df_sp2[df_sp2["category"] == cat].copy()
        if sp1_cat.empty and sp2_cat.empty:
            continue
        matches = two_pass_match(sp1_cat, sp2_cat, cat)
        for m in matches:
            r1, r2 = m["r1"], m["r2"]
            compare_rows.append({
                **m,
                "category"       : cat,
                "SP1_serial"     : r1["serial_no"]    if r1 is not None else "",
                "SP1_description": r1["display_desc"] if r1 is not None else "",
                "SP2_serial"     : r2["serial_no"]    if r2 is not None else "",
                "SP2_description": r2["display_desc"] if r2 is not None else "",
            })

    console.print()

    # Build a rich comparison table for one category
    def make_compare_table(cat_matches, sp1_title, sp2_title):
        tbl = Table(
            box          = box.SIMPLE_HEAVY,
            show_lines   = True,
            expand       = True,
            padding      = (0, 1),
            header_style = "bold white on #37474f",
            show_header  = True,
        )
        tbl.add_column("Status",                    style="bold #000000", width=14, no_wrap=False)
        tbl.add_column(f"SP1 Serial\n{sp1_title}",  style="bold #005500", width=10, no_wrap=True)
        tbl.add_column("SP1 Description / SQL / Details", style="#000000", overflow="fold", ratio=1)
        tbl.add_column(f"SP2 Serial\n{sp2_title}",  style="bold #003399", width=10, no_wrap=True)
        tbl.add_column("SP2 Description / SQL / Details", style="#000000", overflow="fold", ratio=1)

        for m in cat_matches:
            status = m["status"]
            rs     = STATUS_BG[status]
            label  = STATUS_LABEL[status]
            if status == MODIFIED and m["diffs"]:
                label += f"\n  changed: {', '.join(m['diffs'])}"
            if m["match_type"] == "sql":
                label += "\n  (SQL match)"
            tbl.add_row(
                label,
                m["SP1_serial"], m["SP1_description"],
                m["SP2_serial"], m["SP2_description"],
                style=rs,
            )
        return tbl

    sp1_hdr = f"{outer1.get('title','?')} [{SPACE_ID_1}]"
    sp2_hdr = f"{outer2.get('title','?')} [{SPACE_ID_2}]"

    for cat in CATEGORIES:
        cat_matches = [r for r in compare_rows if r["category"] == cat]
        if not cat_matches: continue
        n_identical = sum(1 for r in cat_matches if r["status"] == IDENTICAL)
        n_modified  = sum(1 for r in cat_matches if r["status"] == MODIFIED)
        n_sp1       = sum(1 for r in cat_matches if r["status"] == SP1_ONLY)
        n_sp2       = sum(1 for r in cat_matches if r["status"] == SP2_ONLY)
        n_sql_match = sum(1 for r in cat_matches if r["match_type"] == "sql")
        label = CAT_LABELS.get(cat, cat)
        summary = (
            f"  {len(cat_matches)} rows  |  "
            f" both:{n_identical}   modified:{n_modified}  "
            f" sp1-only:{n_sp1}  sp2-only:{n_sp2}"
            + (f"  (sql-matched:{n_sql_match})" if n_sql_match else "")
        )
        console.print()
        console.rule(f"  {label}  {summary}  ", style="#000000")
        console.print(make_compare_table(cat_matches, sp1_hdr, sp2_hdr))

    console.print()
    stbl = Table(
        title        = "Comparison Summary",
        title_style  = "bold #000000",
        box          = box.SIMPLE_HEAVY,
        header_style = "bold white on #37474f",
        show_lines   = False,
        padding      = (0, 2),
    )
    stbl.add_column("Category",    style="#000000", width=20)
    stbl.add_column("SP1 Items",   style="#005500", width=10, justify="right")
    stbl.add_column("SP2 Items",   style="#003399", width=10, justify="right")
    stbl.add_column("Both",        style="#000000", width=10, justify="right")
    stbl.add_column("Modified",    style="#856404", width=12, justify="right")
    stbl.add_column("SP1 Only",    style="#005500", width=12, justify="right")
    stbl.add_column("SP2 Only",    style="#003399", width=12, justify="right")
    stbl.add_column("SQL matched", style="#5f35ae", width=12, justify="right")

    total = {k: 0 for k in ["sp1", "sp2", "both", "mod", "s1", "s2", "sql"]}
    for cat in CATEGORIES:
        cat_matches = [r for r in compare_rows if r["category"] == cat]
        if not cat_matches: continue
        c_sp1  = sum(1 for r in cat_matches if r.get("_in_sp1", r["r1"] is not None) or r["SP1_serial"])
        c_sp2  = sum(1 for r in cat_matches if r.get("_in_sp2", r["r2"] is not None) or r["SP2_serial"])
        c_both = sum(1 for r in cat_matches if r["status"] == IDENTICAL)
        c_mod  = sum(1 for r in cat_matches if r["status"] == MODIFIED)
        c_s1   = sum(1 for r in cat_matches if r["status"] == SP1_ONLY)
        c_s2   = sum(1 for r in cat_matches if r["status"] == SP2_ONLY)

    only1_instr = {k: v for k, v in sp1_instr_map.items() if k not in sp2_instr_map}

# CELL 5 - BUILD MERGED SPACE
#
# Merge logic:
#   1. Take all items from the target space as the base
#   2. Remove any items listed in Widget 6 (from target)
#   3. Add selected items from Widget 5 into their correct sections
#   4. Deduplicate every section - target wins on conflict (same norm_key)
#   5. Instructions: target lines kept in order, source lines appended below
#      with one blank line separating the two blocks
#   6. AUTO FULL MERGE: if Widget 5 and Widget 6 are both blank, merge ALL
#      items from both spaces (Instructions, Tables, Joins, Measures, Expressions,
#      Filters, Benchmarks, Sample Questions, Example SQLs) with no duplicates.

mode = dbutils.widgets.get("mode")

if mode == "merge":

    # Determine which space is target and which is source
    target_choice = dbutils.widgets.get("target_space").strip()
    is_sp1_target = target_choice.startswith("SP1")

    if is_sp1_target:
        target_registry = registry1
        source_registry = registry2
        TARGET_SPACE_ID = SPACE_ID_1
        SOURCE_SPACE_ID = SPACE_ID_2
        outer_target    = outer1
        outer_source    = outer2
        target_ss       = ss1
        source_ss       = ss2
        target_label    = f"Space 1 : {outer1.get('title','?')}"
        source_label    = f"Space 2 : {outer2.get('title','?')}"
        target_prefix   = "SP1"
        source_prefix   = "SP2"
    else:
        target_registry = registry2
        source_registry = registry1
        TARGET_SPACE_ID = SPACE_ID_2
        SOURCE_SPACE_ID = SPACE_ID_1
        outer_target    = outer2
        outer_source    = outer1
        target_ss       = ss2
        source_ss       = ss1
        target_label    = f"Space 2 : {outer2.get('title','?')}"
        source_label    = f"Space 1 : {outer1.get('title','?')}"
        target_prefix   = "SP2"
        source_prefix   = "SP1"

    # Parse comma-separated serial numbers from a widget string
    def parse_serials(raw):
        return set(s.strip().upper() for s in raw.split(",") if s.strip())

    add_serials    = parse_serials(dbutils.widgets.get("serials_to_add"))
    remove_serials = parse_serials(dbutils.widgets.get("serials_to_remove"))

    # AUTO FULL MERGE: if both Widget 5 and Widget 6 are blank, merge everything
    # from both spaces into the target, deduplicating across all components.
    # Target always wins when the same item exists in both spaces.
    _auto_full_merge = (not add_serials) and (not remove_serials)
    if _auto_full_merge:
        console.print(Panel(
            "Widget 5 (serials to add) and Widget 6 (serials to remove) are both blank.\n\n"
            "AUTO FULL MERGE MODE activated:\n"
            "  All items from both spaces are merged into the target space.\n"
            "  This includes: Instructions, Tables, Joins, Measures, Expressions,\n"
            "                 Filters, Benchmarks, Sample Questions, Example SQLs.\n\n"
            "  Duplicates are suppressed in every component - no duplicates anywhere.\n"
            "  Target version wins whenever the same item exists in both spaces.\n\n"
            "To do a selective merge instead, enter serial numbers in Widget 5 / Widget 6.",
            title="AUTO FULL MERGE", border_style="#5f35ae"))

    # Route a serial number to its correct registry based on prefix
    def _resolve_registry(sn):
        if sn.startswith("SP1_"): return registry1, "Space 1"
        if sn.startswith("SP2_"): return registry2, "Space 2"
        return None, None

    # Validate serial numbers and warn about any that are not found
    bad_add = []
    for s in add_serials:
        reg, lbl = _resolve_registry(s)
        if reg is None:
            bad_add.append(f"  {s} - unrecognised prefix (expected SP1_xxx or SP2_xxx)")
        elif s not in reg:
            bad_add.append(f"  {s} - not found in {lbl} registry")

    bad_rem = [f"  {s} - not found in TARGET ({target_prefix})" for s in remove_serials
               if s not in target_registry]

    if bad_add or bad_rem:
        console.print("WARNING - invalid serials (will be ignored):")
        for m in (bad_add + bad_rem): console.print(m)

    # Warn if source-space serials were entered in Widget 6 (remove) - they have no effect
    wrong_rem = [s for s in remove_serials if s.startswith(source_prefix + "_")]
    if wrong_rem:
        console.print(f"\nWARNING: Widget 6 contains {source_prefix} serials from the source space."
                      f" These items are not in the target so removing them has no effect:\n"
                      f"  {', '.join(sorted(wrong_rem))}")

    # Initialize the output merged space structure
    merged_ss = {
        "data_sources": {"tables": []},
        "instructions": {
            "text_instructions"    : [],
            "join_specs"           : [],
            "example_question_sqls": [],
            "sql_snippets"         : {"filters": [], "expressions": [], "measures": []},
        },
        "benchmarks": {"questions": []},
        "config"    : {"sample_questions": []},
        # Always use the target space version so Databricks PATCH does not get a version conflict
        "version"   : target_ss.get("version", 1),
    }
    ds   = merged_ss["data_sources"]
    inst = merged_ss["instructions"]
    snip = inst["sql_snippets"]

    # Track which items have already been written to prevent duplicates (by norm_key)
    seen = {c: set() for c in [
        "TABLE", "JOIN", "MEASURE", "EXPRESSION", "FILTER",
        "BENCHMARK", "SAMPLE_QUESTION", "EXAMPLE_SQL", "INSTRUCTION"
    ]}
    merged_registry = {}
    rpt_base, rpt_added, rpt_removed, rpt_skipped = [], [], [], []

    # Write one registry entry into merged_ss. Returns True if written, False if duplicate.
    # Instructions are tracked in the registry only; actual content is assembled separately below.
    def apply_item(entry, sn):
        cat, key = entry["category"], entry["key"]
        item     = copy.deepcopy(entry["item"])

        if key in seen[cat]:
            rpt_skipped.append(f"  {sn} [{cat}] {str(key)[:80]}  (duplicate - target version kept)")
            return False

        seen[cat].add(key)
        instr_line = None

        if   cat == "TABLE"          : ds["tables"].append(item)
        elif cat == "JOIN"           : inst["join_specs"].append(item)
        elif cat == "MEASURE"        : snip["measures"].append(item)
        elif cat == "EXPRESSION"     : snip["expressions"].append(item)
        elif cat == "FILTER"         : snip["filters"].append(item)
        elif cat == "BENCHMARK"      : merged_ss["benchmarks"]["questions"].append(item)
        elif cat == "SAMPLE_QUESTION": merged_ss["config"]["sample_questions"].append(item)
        elif cat == "EXAMPLE_SQL"    : inst["example_question_sqls"].append(item)
        elif cat == "INSTRUCTION"    :
            instr_line = item["line"]

        merged_registry[sn] = {
            "category"  : cat,
            "key"       : key,
            "item"      : copy.deepcopy(item),
            "instr_line": instr_line,
        }
        return True

    # Step 1: Add all target items as the base, skipping any in the remove list
    for sn in sorted(target_registry.keys()):
        if sn in remove_serials:
            e = target_registry[sn]
            rpt_removed.append(f"  {sn} [{e['category']}] {str(e['key'])[:80]}")
            continue
        e = target_registry[sn]
        if apply_item(e, sn):
            rpt_base.append(f"  {sn} [{e['category']}] {str(e['key'])[:80]}")

    # Step 2: Add items from Widget 5, OR in auto-full-merge mode add ALL source items
    if _auto_full_merge:
        # Add every item from the source space; apply_item deduplicates automatically
        for sn in sorted(source_registry.keys()):
            e = source_registry[sn]
            if apply_item(e, sn):
                rpt_added.append(f"  {sn} [{e['category']}] [auto-merge] {str(e['key'])[:80]}")
    else:
        # Selective add: only the serial numbers entered in Widget 5
        for sn in sorted(add_serials):
            reg, reg_label = _resolve_registry(sn)
            if reg is None or sn not in reg:
                continue
            e = reg[sn]
            if apply_item(e, sn):
                rpt_added.append(f"  {sn} [{e['category']}] [{reg_label}] {str(e['key'])[:80]}")

    # =========================================================================
    # Step 3 – Build the final instruction text block
    #
    # ROOT CAUSE OF CONCATENATION BUG:
    #   The Databricks API stores instructions in text_instructions[0]["content"]
    #   as a SINGLE \n-delimited STRING  (e.g. "Line1\nLine2\nLine3").
    #   Previous versions set content = a Python list, which gets serialised as
    #   a JSON array.  Databricks then joins the array elements with no separator,
    #   so all lines appear on one line inside the Genie UI.
    #
    # FIX:
    #   content must ALWAYS be written back as one plain string with \n between
    #   every instruction line.  We never store a list in content.
    #
    # APPROACH – "lift and shift":
    #   Full merge  : take the target's raw content string as-is, then append
    #                 the source's raw content string as-is (blank line between).
    #                 Deduplicate line-by-line (target wins); join with \n.
    #   Selective   : take the target's raw content string as-is, then append
    #                 only the lines that correspond to the selected serial numbers,
    #                 one line per serial, joined with \n.
    #   Removals    : lines whose norm_key matches a Widget-6 serial are dropped
    #                 from the target block before appending anything.
    # =========================================================================

    def _raw_content_to_str(content):
        """
        Normalise the raw 'content' value from the API into a plain string.
        The API may return a str or occasionally a list.
        Always returns a str (lines separated by \\n).
        """
        if isinstance(content, str):
            return content
        if isinstance(content, list):
            # Flatten any nested newlines and join with \n
            parts = []
            for entry in content:
                for ln in str(entry).split("\n"):
                    parts.append(ln)
            return "\n".join(parts)
        return ""

    def _str_to_lines(s):
        """Split a \n-separated instruction string into individual lines."""
        return s.split("\n") if s else []

    # ---- Collect norm_keys of instructions the user wants removed (Widget 6) ----
    _rem_instr_nkeys = {
        norm_key(target_registry[sn]["key"])
        for sn in remove_serials
        if sn in target_registry and target_registry[sn]["category"] == "INSTRUCTION"
    }

    # ---- Deep-copy the target's text_instructions block (keeps id and other API fields) ----
    _raw_ti_list  = copy.deepcopy(get_txt(target_ss))
    _ti_shell     = _raw_ti_list[0] if _raw_ti_list else {}
    _raw_content  = _ti_shell.get("content", "")

    # Convert whatever the API returned into a clean string, then into a line list
    _target_all_lines = _str_to_lines(_raw_content_to_str(_raw_content))

    # ---- Filter out removed lines; track which norm_keys are already present ----
    _existing_nkeys = set()
    _kept_lines     = []
    for _ln in _target_all_lines:
        _nk = norm_key(_ln)
        if _nk in _rem_instr_nkeys:
            continue          # user explicitly removed this line
        _existing_nkeys.add(_nk)
        _kept_lines.append(_ln)

    # ---- Collect lines to append from the source ----
    _lines_added = []

    if _auto_full_merge:
        # FULL MERGE – lift-and-shift: grab the entire source content block,
        # iterate its lines in original order, skip duplicates.
        _src_ti_list   = get_txt(source_ss)
        _src_ti_shell  = _src_ti_list[0] if _src_ti_list else {}
        _src_raw       = _src_ti_shell.get("content", "")
        _src_all_lines = _str_to_lines(_raw_content_to_str(_src_raw))

        for _ln in _src_all_lines:
            _nk = norm_key(_ln)
            if _nk not in _existing_nkeys:
                _existing_nkeys.add(_nk)
                _lines_added.append(_ln)
            # If the line already exists in the target we simply skip it (target wins)

    else:
        # SELECTIVE – add only the lines that correspond to Widget-5 serial numbers.
        # Serial numbers are processed in the ORDER the user typed them so that the
        # appended block respects the user's intended sequence.
        _ordered_add_serials = [
            s.strip().upper()
            for s in dbutils.widgets.get("serials_to_add").split(",")
            if s.strip()
        ]
        for sn in _ordered_add_serials:
            reg, _ = _resolve_registry(sn)
            if reg is None or sn not in reg:
                continue
            e = reg[sn]
            if e["category"] != "INSTRUCTION":
                continue
            _line_text = e["item"]["line"]
            _nk = norm_key(_line_text)
            if _nk not in _existing_nkeys:
                _existing_nkeys.add(_nk)
                _lines_added.append(_line_text)

    # ---- Assemble the final content string ----
    # One blank line separates the target block from the appended block.
    # Every instruction occupies exactly one line – no concatenation possible
    # because we write a single \n-joined string (not a list).
    if _kept_lines and _lines_added:
        _final_content_str = "\n".join(_kept_lines) + "\n\n" + "\n".join(_lines_added)
    elif _kept_lines:
        _final_content_str = "\n".join(_kept_lines)
    elif _lines_added:
        _final_content_str = "\n".join(_lines_added)
    else:
        _final_content_str = ""

    # ---- Write back into the merged space ----
    # content is stored as a plain string – exactly what the Databricks API expects.
    if _final_content_str or _ti_shell:
        _ti_shell["content"] = _final_content_str   # single \n-delimited string
        inst["text_instructions"] = [_ti_shell]

    # Counts for the summary report
    _n_target_kept = len([l for l in _kept_lines if l.strip()])
    _n_removed     = len(_rem_instr_nkeys)
    _n_added       = len([l for l in _lines_added if l.strip()])

    # Print merge summary
    _merge_mode_label = "AUTO FULL MERGE" if _auto_full_merge else "Selective Merge"
    body = (
        f"Mode    : {_merge_mode_label}\n"
        f"Target  : {target_label}  [{TARGET_SPACE_ID}]\n"
        f"Source  : {source_label}  [{SOURCE_SPACE_ID}]\n\n"
        f"Base items from target kept : {len(rpt_base)}\n"
        f"Items removed from target   : {len(rpt_removed)}\n"
        f"Items added from source     : {len(rpt_added)}\n"
        f"Duplicates suppressed       : {len(rpt_skipped)}"
    )
    if rpt_removed:
        body += "\n\nRemoved from target:\n" + "\n".join(rpt_removed)
    if rpt_added:
        body += "\n\nAdded from source:\n" + "\n".join(rpt_added)
    if rpt_skipped:
        body += "\n\nSkipped (duplicate key, target version kept):\n" + "\n".join(rpt_skipped)

    console.print(Panel(body, title=f"Merged Space Built  [{_merge_mode_label}]", border_style="#005500"))
    console.print(f"  Tables           : {len(ds['tables'])}")
    console.print(f"  Joins            : {len(inst['join_specs'])}")
    _merged_ti   = inst.get("text_instructions", [])
    _ti_content  = _merged_ti[0].get("content", "") if _merged_ti else ""
    _total_lines = len([l for l in _ti_content.split("\n") if l.strip()]) if isinstance(_ti_content, str) else 0
    console.print(f"  Instructions     : {_n_target_kept} kept + {_n_added} added  ({_n_removed} removed)  = {_total_lines} lines total")
    console.print(f"  Measures         : {len(snip['measures'])}")
    console.print(f"  Expressions      : {len(snip['expressions'])}")
    console.print(f"  Filters          : {len(snip['filters'])}")
    console.print(f"  Benchmarks       : {len(merged_ss['benchmarks']['questions'])}")
    console.print(f"  Sample Questions : {len(merged_ss['config']['sample_questions'])}")
    console.print(f"  Example SQLs     : {len(inst['example_question_sqls'])}")
    console.print(f"  JSON size        : {len(json.dumps(merged_ss, ensure_ascii=False)):,} chars")

    sp2_merged_ss = merged_ss
    sp2_filtered  = False

    console.print("\nRun Cell 6 to clean, sort, and validate.")

else:
    console.print("Mode = rollback - Cell 5 skipped.")


# CELL 6 - CLEAN + SORT + VALIDATE
#
# FIX: reassign_duplicate_ids() replaces duplicate 'id' fields with fresh UUIDs
# instead of dropping the record.
#
# Root cause of HTTP 400 "Duplicate question ID" error:
#   Two spaces can have items (benchmarks, joins, example SQLs, measures, etc.)
#   with DIFFERENT content but the SAME internal UUID 'id' field.
#   The norm_key dedup in Cell 5 only catches same-content duplicates, so a
#   same-id different-content pair both survive. Databricks API then returns:
#     HTTP 400 "Duplicate question ID '<uuid>': first seen in
#     benchmarks.questions[N], duplicated in benchmarks.questions[M]"
#
# Fix strategy (per user requirement):
#   - Target items are always written first (Step 1 in Cell 5).
#   - Source items are appended after (Step 2 in Cell 5).
#   - When a duplicate 'id' is found it always belongs to a source-space item.
#   - Instead of dropping it: generate a brand-new UUID and assign it as the
#     item's 'id'. The item (with all its content) is KEPT in full.
#   - The ID is ONLY reassigned in the final merged structure (ready_ss),
#     never in the original registry or the raw space data.
#   - After this pass every item has a unique 'id', zero records are lost,
#     and zero duplicate IDs remain anywhere.

mode = dbutils.widgets.get("mode")

if mode == "merge":

    # Fix description fields - normalize to a clean list or remove if empty
    def fix_desc(v):
        if isinstance(v, list):
            c = [str(x) for x in v if x is not None and str(x).strip()]
            return c if c else None
        if isinstance(v, str) and v.strip(): return [v.strip()]
        return None

    # Sanitize table and column description fields
    def sanitize(ss):
        for t in ss.get("data_sources", {}).get("tables", []):
            if not isinstance(t, dict): continue
            if "description" in t:
                f = fix_desc(t["description"])
                if f: t["description"] = f
                else: del t["description"]
            for c in t.get("column_configs", []):
                if not isinstance(c, dict): continue
                if "description" in c:
                    f = fix_desc(c["description"])
                    if f: c["description"] = f
                    else: del c["description"]

    # =========================================================================
    # CORE BUG FIX: Resolve duplicate 'id' fields by reassigning fresh UUIDs.
    #
    # Problem: Two spaces can have items (benchmarks, joins, example SQLs,
    # measures, etc.) with DIFFERENT content but the SAME internal UUID 'id'.
    # The norm_key dedup in Cell 5 only catches same-content duplicates, so a
    # same-id different-content pair both survive into merged_ss.
    # Databricks API then returns HTTP 400 "Duplicate question ID '<uuid>'".
    #
    # Fix (per requirement):
    #   - Do NOT drop any record.
    #   - The first occurrence of each 'id' always belongs to the TARGET space
    #     (target items are written before source items in Cell 5).
    #   - Every SUBSEQUENT occurrence (source-space item) gets a brand-new UUID
    #     generated by uuid.uuid4().hex  ->  32 lowercase hex chars, no dashes,
    #     matching Databricks' own ID format exactly.
    #   - IDs are reassigned ONLY inside ready_ss (the final merged copy).
    #     Original registry data and raw space dicts are never touched.
    #   - After this pass: every item has a unique 'id', no records lost,
    #     no duplicate IDs anywhere in any section.
    # =========================================================================
    def reassign_duplicate_ids(ss, source_prefix):
        """
        Walk every list in the merged serialized space.
        For any item whose 'id' is already taken by an earlier item, generate a
        new UUID and assign it.  All items are kept - nothing is dropped.

        Parameters
        ----------
        ss            : the merged serialized-space dict (ready_ss)
        source_prefix : 'SP1' or 'SP2' - used only for logging clarity
        """

        # Accumulate all IDs that have been seen so far across ALL sections
        # so that a newly generated ID never accidentally collides with an
        # existing ID in any other section.
        all_known_ids = set()

        def _collect_existing_ids(lst):
            """First pass: collect all ids already present in a list."""
            for item in lst:
                if isinstance(item, dict):
                    iid = item.get("id")
                    if iid:
                        all_known_ids.add(iid)

        def _new_uuid():
            """Generate a fresh 32-char hex UUID that is not already in use."""
            while True:
                candidate = uuid.uuid4().hex          # e.g. 'a1b2c3d4e5f6...'
                if candidate not in all_known_ids:
                    all_known_ids.add(candidate)
                    return candidate

        def _reassign_list(lst, section_label):
            """
            Second pass: for any item whose 'id' was already seen,
            assign a brand-new UUID.  Return the (possibly modified) list
            and a report of what was changed.
            """
            seen_in_section = set()
            reassigned = []
            for item in lst:
                if not isinstance(item, dict):
                    continue
                item_id = item.get("id")
                if item_id is None:
                    # No id field at all - nothing to check
                    continue
                if item_id in seen_in_section:
                    # Duplicate within this section (rare but possible)
                    new_id = _new_uuid()
                    old_id = item["id"]
                    item["id"] = new_id
                    reassigned.append(
                        f"    section={section_label}  "
                        f"old_id={old_id}  ->  new_id={new_id}  "
                        f"[intra-section dup from {source_prefix}]"
                    )
                elif item_id in all_known_ids and item_id not in seen_in_section:
                    # This id was seen in a PREVIOUS section or is a cross-section dup
                    # We only reassign if the id would actually clash in THIS section.
                    # Cross-section duplicates are not rejected by Databricks, so we
                    # only need to fix duplicates within the SAME section list.
                    # (Databricks validates uniqueness per-section, not globally.)
                    pass
                seen_in_section.add(item_id)
            return reassigned

        # -----------------------------------------------------------------------
        # We scan section by section.
        # Within each section: first occurrence (= target item) keeps its id;
        # any later occurrence (= source item with same id) gets a new UUID.
        # -----------------------------------------------------------------------
        def _process_section(lst, section_label):
            seen_ids  = set()
            reassigned_log = []
            for item in lst:
                if not isinstance(item, dict):
                    continue
                item_id = item.get("id")
                if item_id is None:
                    continue
                if item_id in seen_ids:
                    # This IS the duplicate - it came from the source space.
                    # Reassign a new UUID so both records survive.
                    new_id = _new_uuid()
                    old_id = item_id
                    item["id"] = new_id
                    reassigned_log.append(
                        f"    [{section_label}]  "
                        f"old_id={old_id}  ->  new_id={new_id}  "
                        f"(source-space item kept with new UUID)"
                    )
                else:
                    seen_ids.add(item_id)
                    all_known_ids.add(item_id)
            return reassigned_log

        ds   = ss.get("data_sources",  {})
        inst = ss.get("instructions",  {})
        snip = inst.get("sql_snippets", {})
        bm   = ss.get("benchmarks",    {})
        cfg  = ss.get("config",        {})

        all_rpt = []

        if "tables" in ds:
            all_rpt += _process_section(ds["tables"], "data_sources.tables")

        if "join_specs" in inst:
            all_rpt += _process_section(
                inst["join_specs"], "instructions.join_specs")

        if "example_question_sqls" in inst:
            all_rpt += _process_section(
                inst["example_question_sqls"],
                "instructions.example_question_sqls")

        if "text_instructions" in inst:
            all_rpt += _process_section(
                inst["text_instructions"], "instructions.text_instructions")

        for k in ("filters", "expressions", "measures"):
            if k in snip:
                all_rpt += _process_section(snip[k], f"sql_snippets.{k}")

        # -----------------------------------------------------------------
        # PRIMARY FIX: benchmarks.questions
        # This is where the HTTP 400 "Duplicate question ID" error originates.
        # -----------------------------------------------------------------
        if isinstance(bm, dict) and "questions" in bm:
            all_rpt += _process_section(
                bm["questions"], "benchmarks.questions")

        if "sample_questions" in cfg:
            all_rpt += _process_section(
                cfg["sample_questions"], "config.sample_questions")

        # Print a summary of all ID reassignments
        if all_rpt:
            console.print(Panel(
                f"  {len(all_rpt)} duplicate ID(s) detected.\n"
                f"  Each was given a brand-new UUID so NO record is dropped.\n"
                f"  The target-space item always keeps its original ID.\n"
                f"  The source-space item ({source_prefix}) receives the new UUID.\n\n"
                + "\n".join(all_rpt),
                title="ID Reassignment Report (duplicate IDs resolved)",
                border_style="#856404"))
        else:
            console.print("  [reassign_duplicate_ids] No duplicate IDs found - all sections are clean.")

    # Sort all sections by their id field (required by the Databricks API)
    def sort_all(ss):
        by_id    = lambda l: sorted(l, key=lambda x: x.get("id", "z" * 32) if isinstance(x, dict) else "z" * 32)
        by_ident = lambda l: sorted(l, key=lambda x: x.get("identifier", "") if isinstance(x, dict) else "")
        ds, inst = ss.get("data_sources", {}), ss.get("instructions", {})
        snip = inst.get("sql_snippets", {})
        bm   = ss.get("benchmarks", {})
        cfg  = ss.get("config", {})
        if "tables"       in ds: ds["tables"]       = by_ident(ds["tables"])
        if "metric_views" in ds: ds["metric_views"]  = by_ident(ds["metric_views"])
        if "join_specs"            in inst: inst["join_specs"]            = by_id(inst["join_specs"])
        if "example_question_sqls" in inst: inst["example_question_sqls"] = by_id(inst["example_question_sqls"])
        if "text_instructions"     in inst: inst["text_instructions"]     = by_id(inst["text_instructions"])
        for k in ("filters", "expressions", "measures"):
            if k in snip: snip[k] = by_id(snip[k])
        if isinstance(bm, dict) and "questions" in bm: bm["questions"] = by_id(bm["questions"])
        if "sample_questions" in cfg: cfg["sample_questions"] = by_id(cfg["sample_questions"])

    ready_ss = copy.deepcopy(merged_ss)
    sanitize(ready_ss)

    # Resolve duplicate IDs BEFORE sorting.
    # Operates only on ready_ss (the final merged copy) - originals untouched.
    # Source-space items that share an 'id' with a target-space item are kept
    # in full; only their 'id' field is replaced with a fresh UUID.
    console.print("  Running reassign_duplicate_ids pass ...")
    reassign_duplicate_ids(ready_ss, source_prefix)

    sort_all(ready_ss)

    # Validate that the output is valid JSON
    try:
        s = json.dumps(ready_ss, ensure_ascii=False)
        json.loads(s)
        console.print(f"  ready_ss - valid JSON  {len(s):,} chars")
    except Exception as e:
        console.print(f"  ready_ss JSON validation failed: {e}"); raise

    ready_sp2_ss = ready_ss

    _inst  = ready_ss.get("instructions", {})
    _snip  = _inst.get("sql_snippets", {})
    _bm    = ready_ss.get("benchmarks", {})

    def _ids(lst): return [x.get("id", "") for x in lst if isinstance(x, dict)]

    # Check that a list is sorted by id and print the result
    def _chk(lst, label):
        ids = _ids(lst)
        ok  = ids == sorted(ids)
        status = "sorted" if ok else "NOT SORTED - API will reject"
        console.print(f"  {label:<28} : {len(lst):>4} items   {status}")

    # Check for any remaining duplicate ids (should be zero after dedup_by_id)
    def _chk_dup_ids(lst, label):
        ids = [x.get("id") for x in lst if isinstance(x, dict) and x.get("id")]
        dups = {i for i in ids if ids.count(i) > 1}
        if dups:
            console.print(f"  WARNING duplicate ids still present in {label}: {dups}")
        else:
            console.print(f"  {label:<28} : no duplicate ids")

    txt = len(_inst.get("text_instructions", []))
    console.print(f"  text_instructions blocks : {txt}  (expected 1 or less)")
    _chk(_inst.get("join_specs", []),            "join_specs")
    _chk(_inst.get("example_question_sqls", []), "example_question_sqls")
    _chk(_inst.get("text_instructions", []),     "text_instructions")
    _chk(_snip.get("filters", []),               "filters")
    _chk(_snip.get("expressions", []),           "expressions")
    _chk(_snip.get("measures", []),              "measures")
    _chk(_bm.get("questions", []) if isinstance(_bm, dict) else [], "benchmarks.questions")
    _chk(ready_ss.get("config", {}).get("sample_questions", []),    "sample_questions")

    console.print()
    console.print("  Duplicate-id checks (should all be zero after reassign pass):")
    _chk_dup_ids(_inst.get("join_specs", []),            "join_specs")
    _chk_dup_ids(_inst.get("example_question_sqls", []), "example_question_sqls")
    _chk_dup_ids(_snip.get("filters", []),               "filters")
    _chk_dup_ids(_snip.get("expressions", []),           "expressions")
    _chk_dup_ids(_snip.get("measures", []),              "measures")
    _chk_dup_ids(_bm.get("questions", []) if isinstance(_bm, dict) else [], "benchmarks.questions")
    _chk_dup_ids(ready_ss.get("config", {}).get("sample_questions", []), "sample_questions")

    console.print("\nAll sort and dedup checks done. Run Cell 7 to snapshot, export JSON, and generate PDF.")

else:
    console.print("Mode = rollback - Cell 6 skipped.")

        f"Target (will be patched)  : {target_label}  [{TARGET_SPACE_ID}]\n"
        f"Source (unchanged)        : {source_label}  [{SOURCE_SPACE_ID}]\n\n"
        f"Items in merged space:\n"
        f"  Tables           : {len(ready_ss.get('data_sources',{}).get('tables',[]))}\n"
        f"  Joins            : {len(ready_ss.get('instructions',{}).get('join_specs',[]))}\n"
        f"  Measures         : {len(ready_ss.get('instructions',{}).get('sql_snippets',{}).get('measures',[]))}\n"
        f"  Expressions      : {len(ready_ss.get('instructions',{}).get('sql_snippets',{}).get('expressions',[]))}\n"
        f"  Filters          : {len(ready_ss.get('instructions',{}).get('sql_snippets',{}).get('filters',[]))}\n"
        f"  Example SQLs     : {len(ready_ss.get('instructions',{}).get('example_question_sqls',[]))}\n"
        f"  Benchmarks       : {len(ready_ss.get('benchmarks',{}).get('questions',[]))}\n"
        f"  Sample Questions : {len(ready_ss.get('config',{}).get('sample_questions',[]))}",
        title="Cell 9 - Patch Target Space", border_style="yellow"))

    # Fetch the current live version just before patching.
    # Databricks uses optimistic concurrency - version must match exactly.
    # Any re-run or concurrent edit since Cell 2 would cause a 409 conflict without this refresh.
    _ver_r = requests.get(
        f"{DATABRICKS_HOST}/api/2.0/genie/spaces/{TARGET_SPACE_ID}",
        headers=HEADERS,
        params={"include_serialized_space": "true"},
        timeout=30,
    )
    if _ver_r.status_code == 200:
        _ver_raw = _ver_r.json().get("serialized_space", "{}")
        _live_ver = (json.loads(_ver_raw) if isinstance(_ver_raw, str) else _ver_raw).get("version", 1)
        ready_ss["version"] = _live_ver
        console.print(f"  Live target version refreshed: {_live_ver}")
    else:
        console.print(f"  Could not refresh target version (HTTP {_ver_r.status_code}) - proceeding with cached version")

    resp_p = requests.patch(
        f"{DATABRICKS_HOST}/api/2.0/genie/spaces/{TARGET_SPACE_ID}",
        headers=HEADERS,
        json={
            "title"           : outer_target["title"],
            "description"     : outer_target.get("description", ""),
            "warehouse_id"    : outer_target["warehouse_id"],
            "serialized_space": json.dumps(ready_ss, ensure_ascii=False),
        },
        timeout=120,
    )

    if resp_p.status_code == 200:
        console.print(Panel(
            f"Target space patched successfully.\n\n"
            f"  Patched   : '{outer_target['title']}'  [{TARGET_SPACE_ID}]\n"
            f"  Untouched : '{outer_source['title']}'  [{SOURCE_SPACE_ID}]\n\n"
            f"To rollback:\n"
            f"  1. Set mode -> rollback\n"
            f"  2. Enter snapshot -> {snap1_name if is_sp1_target else snap2_name}\n"
            f"  3. Run Cell 10",
            title="Cell 9 - Patch complete", border_style="#005500"))
    else:
        console.print(f"  Error  HTTP {resp_p.status_code}")
        console.print(f"  {resp_p.text[:800]}")
        if resp_p.status_code == 409:
            console.print("  409 = version conflict. The space was modified after the version refresh.")
            console.print("  Re-run from Cell 2 to re-fetch, then retry Cell 9.")
        else:
            console.print("  Check CAN EDIT permission on the target space.")

else:
    console.print("Mode = rollback - Cell 9 skipped.")


# CELL 10 - ROLLBACK

mode = dbutils.widgets.get("mode")

if mode == "rollback":

    rb_file = dbutils.widgets.get("rollback_file").strip()
    if not rb_file:
        console.print("Enter snapshot filename in Widget 2.")
        raise ValueError("rollback_file widget is empty.")

    snap_path = f"{SNAPSHOT_FOLDER}/{rb_file}"
    console.print(f"Loading : {snap_path}")

    try:
        with open(snap_path, "r", encoding="utf-8") as f:
            rb = json.load(f)
        console.print("  Snapshot loaded.")
    except FileNotFoundError:
        console.print(f"  Not found: {snap_path}"); raise
    except Exception as e:
        console.print(f"  Load error: {e}"); raise

    sid  = rb.get("space_id", "")
    raw  = rb.get("serialized_space", "{}")
    ss_r = json.loads(raw) if isinstance(raw, str) else raw
    ds_r = ss_r.get("data_sources", {})
    in_r = ss_r.get("instructions", {})
    sn_r = in_r.get("sql_snippets", {})
    bm_r = ss_r.get("benchmarks", {})
    bmc  = (len(bm_r.get("questions", [])) if isinstance(bm_r, dict)
            else len(bm_r) if isinstance(bm_r, list) else 0)

    console.print(Panel(
        f"Title        : '{rb.get('title','?')}'\nSpace ID     : {sid}\n"
        f"Tables       : {len(ds_r.get('tables',[]))}\n"
        f"Joins        : {len(in_r.get('join_specs',[]))}\n"
        f"Example SQLs : {len(in_r.get('example_question_sqls',[]))}\n"
        f"Measures     : {len(sn_r.get('measures',[]))}\n"
        f"Expressions  : {len(sn_r.get('expressions',[]))}\n"
        f"Filters      : {len(sn_r.get('filters',[]))}\n"
        f"Benchmarks   : {bmc}\n"
        f"Sample Qs    : {len(ss_r.get('config',{}).get('sample_questions',[]))}",
        title="Restoring snapshot", border_style="yellow"))

    # Fetch the current live version before patching.
    # The snapshot version is stale after a successful merge patch, so we inject
    # the current live version into ss_r to avoid a 409 version conflict.
    _rb_ver_r = requests.get(
        f"{DATABRICKS_HOST}/api/2.0/genie/spaces/{sid}",
        headers=HEADERS,
        params={"include_serialized_space": "true"},
        timeout=30,
    )
    if _rb_ver_r.status_code == 200:
        _rb_raw = _rb_ver_r.json().get("serialized_space", "{}")
        _rb_live_ver = (json.loads(_rb_raw) if isinstance(_rb_raw, str) else _rb_raw).get("version", 1)
        ss_r["version"] = _rb_live_ver
        console.print(f"  Live space version refreshed: {_rb_live_ver}  (snapshot version replaced)")
    else:
        console.print(f"  Could not refresh live version (HTTP {_rb_ver_r.status_code}) - rollback may get 409")

    resp_rb = requests.patch(
        f"{DATABRICKS_HOST}/api/2.0/genie/spaces/{sid}",
        headers=HEADERS,
        json={
            "title"           : rb["title"],
            "description"     : rb.get("description", ""),
            "warehouse_id"    : rb["warehouse_id"],
            # Use ss_r (not the raw snapshot string) so the refreshed live version is included
            "serialized_space": json.dumps(ss_r, ensure_ascii=False),
        },
        timeout=120,
    )

    if resp_rb.status_code == 200:
        console.print(Panel(
            f"Rollback successful.\n"
            f"  '{rb['title']}'  [{sid}]  restored to pre-merge state.\n"
            "  Open the space in Databricks to verify.",
            title="Cell 10 - Rollback complete", border_style="#005500"))
    else:
        console.print(f"  HTTP {resp_rb.status_code}")
        console.print(f"  {resp_rb.text[:600]}")
        if resp_rb.status_code == 409:
            console.print("  409 = version conflict - space was modified after the version refresh.")
            console.print("  Wait a moment and re-run Cell 10 to pick up the latest version.")
        else:
            console.print("  Check CAN EDIT permission on the target space.")

else:
    console.print("Mode = merge - Cell 10 (rollback) skipped.")
