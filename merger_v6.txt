    # Normalize SQL for comparison
    def norm_sql(val):
        if val is None:
            return ""
        if isinstance(val, list):
            val = "".join(str(x) for x in val)
        s = unicodedata.normalize("NFKC", str(val))
        s = re.sub(r"\s+", " ", s)
        return s.strip().casefold()

    # Check whether two matched rows are identical or modified
    def content_status(r1, r2):
        sql_match  = norm_sql(r1["sql"])         == norm_sql(r2["sql"])
        desc_match = norm_key(r1["description"]) == norm_key(r2["description"])
        name_match = norm_key(r1["name"])        == norm_key(r2["name"])
        if sql_match and desc_match and name_match:
            return IDENTICAL, []
        diffs = []
        if not name_match: diffs.append("name")
        if not sql_match:  diffs.append("SQL")
        if not desc_match: diffs.append("description")
        return MODIFIED, diffs

    # Match items from two spaces - first by name, then by SQL for SQL categories
    def two_pass_match(sp1_df, sp2_df, cat):
        results = []

        def collect(df):
            items = []
            for _, r in df.iterrows():
                nk = r["norm_key"]
                sk = norm_sql(r["sql"]) if cat in SQL_CATS else ""
                items.append({"nk": nk, "sk": sk, "row": r, "matched": False})
            return items

        sp1_items = collect(sp1_df)
        sp2_items = collect(sp2_df)

        # Pass 1: match by name

            for i2, e2 in enumerate(sp2_items):
                if e1["matched"] or e2["matched"]:
                    continue
                if e1["nk"] == e2["nk"]:
                    status, diffs = content_status(e1["row"], e2["row"])
                    results.append({
                        "r1": e1["row"], "r2": e2["row"],
                        "status": status, "diffs": diffs,
                        "match_type": "name", "dup_flag": False,

                            "status": status, "diffs": diffs,
                            "match_type": "sql", "dup_flag": False,

# CELL 5 - BUILD MERGED SPACE
#
# Merge logic:
#   1. Take all items from the target space as the base
#   2. Remove any items listed in Widget 6 (from target)
#   3. Add selected items from Widget 5 into their correct sections
#   4. Deduplicate every section - target wins on conflict (same norm_key)
#   5. Instructions: target lines kept in order, source lines appended below
#      with one blank line separating the two blocks

mode = dbutils.widgets.get("mode")

if mode == "merge":

    # Determine which space is target and which is source
    target_choice = dbutils.widgets.get("target_space").strip()
    is_sp1_target = target_choice.startswith("SP1")

    if is_sp1_target:
        target_registry = registry1
        source_registry = registry2
        TARGET_SPACE_ID = SPACE_ID_1
        SOURCE_SPACE_ID = SPACE_ID_2
        outer_target    = outer1
        outer_source    = outer2
        target_ss       = ss1
        source_ss       = ss2
        target_label    = f"Space 1 : {outer1.get('title','?')}"
        source_label    = f"Space 2 : {outer2.get('title','?')}"
        target_prefix   = "SP1"
        source_prefix   = "SP2"
    else:
        target_registry = registry2
        source_registry = registry1
        TARGET_SPACE_ID = SPACE_ID_2
        SOURCE_SPACE_ID = SPACE_ID_1
        outer_target    = outer2
        outer_source    = outer1
        target_ss       = ss2
        source_ss       = ss1
        target_label    = f"Space 2 : {outer2.get('title','?')}"
        source_label    = f"Space 1 : {outer1.get('title','?')}"
        target_prefix   = "SP2"
        source_prefix   = "SP1"

    # Parse comma-separated serial numbers from a widget string
    def parse_serials(raw):
        return set(s.strip().upper() for s in raw.split(",") if s.strip())

    add_serials    = parse_serials(dbutils.widgets.get("serials_to_add"))
    remove_serials = parse_serials(dbutils.widgets.get("serials_to_remove"))

    # AUTO FULL MERGE: if both Widget 5 and Widget 6 are blank, merge everything
    # from both spaces into the target, deduplicating across all components.
    # Target always wins when the same item exists in both spaces.
    _auto_full_merge = (not add_serials) and (not remove_serials)
    if _auto_full_merge:
        console.print(Panel(
            "Widget 5 (serials to add) and Widget 6 (serials to remove) are both blank.\n\n"
            "AUTO FULL MERGE MODE activated:\n"
            "  All items from both spaces are merged into the target space.\n"
            "  This includes: Instructions, Tables, Joins, Measures, Expressions,\n"
            "                 Filters, Benchmarks, Sample Questions, Example SQLs.\n\n"
            "  Duplicates are suppressed in every component - no duplicates anywhere.\n"
            "  Target version wins whenever the same item exists in both spaces.\n\n"
            "To do a selective merge instead, enter serial numbers in Widget 5 / Widget 6.",
            title="AUTO FULL MERGE", border_style="#5f35ae"))

    # Route a serial number to its correct registry based on prefix
    def _resolve_registry(sn):
        if sn.startswith("SP1_"): return registry1, "Space 1"
        if sn.startswith("SP2_"): return registry2, "Space 2"
        return None, None

    # Validate serial numbers and warn about any that are not found
    bad_add = []
    for s in add_serials:
        reg, lbl = _resolve_registry(s)
        if reg is None:
            bad_add.append(f"  {s} - unrecognised prefix (expected SP1_xxx or SP2_xxx)")
        elif s not in reg:
            bad_add.append(f"  {s} - not found in {lbl} registry")

    bad_rem = [f"  {s} - not found in TARGET ({target_prefix})" for s in remove_serials
               if s not in target_registry]

    if bad_add or bad_rem:
        console.print("WARNING - invalid serials (will be ignored):")
        for m in (bad_add + bad_rem): console.print(m)

    # Warn if source-space serials were entered in Widget 6 (remove) - they have no effect
    wrong_rem = [s for s in remove_serials if s.startswith(source_prefix + "_")]
    if wrong_rem:
        console.print(f"\nWARNING: Widget 6 contains {source_prefix} serials from the source space."
                      f" These items are not in the target so removing them has no effect:\n"
                      f"  {', '.join(sorted(wrong_rem))}")

    # Initialize the output merged space structure
    merged_ss = {
        "data_sources": {"tables": []},
        "instructions": {
            "text_instructions"    : [],
            "join_specs"           : [],
            "example_question_sqls": [],
            "sql_snippets"         : {"filters": [], "expressions": [], "measures": []},
        },
        "benchmarks": {"questions": []},
        "config"    : {"sample_questions": []},
        # Always use the target space version so Databricks PATCH does not get a version conflict
        "version"   : target_ss.get("version", 1),
    }

    inst = merged_ss["instructions"]
    snip = inst["sql_snippets"]

    seen = {c: set() for c in [
        "TABLE", "JOIN", "MEASURE", "EXPRESSION", "FILTER",
        "BENCHMARK", "SAMPLE_QUESTION", "EXAMPLE_SQL", "INSTRUCTION"
    ]}
    merged_registry = {}
    rpt_base, rpt_added, rpt_removed, rpt_skipped = [], [], [], []

    # Write one registry entry into merged_ss. Returns True if written, False if duplicate.
    # Instructions are tracked in the registry only; actual content is assembled separately below.
    def apply_item(entry, sn):
        cat, key = entry["category"], entry["key"]
        item     = copy.deepcopy(entry["item"])

        if key in seen[cat]:
            rpt_skipped.append(f"  {sn} [{cat}] {str(key)[:80]}  (duplicate - target version kept)")
            return False

        seen[cat].add(key)
        instr_line = None

        if   cat == "TABLE"          : ds["tables"].append(item)
        elif cat == "JOIN"           : inst["join_specs"].append(item)
        elif cat == "MEASURE"        : snip["measures"].append(item)
        elif cat == "EXPRESSION"     : snip["expressions"].append(item)
        elif cat == "FILTER"         : snip["filters"].append(item)
        elif cat == "BENCHMARK"      : merged_ss["benchmarks"]["questions"].append(item)
        elif cat == "SAMPLE_QUESTION": merged_ss["config"]["sample_questions"].append(item)
        elif cat == "EXAMPLE_SQL"    : inst["example_question_sqls"].append(item)
        elif cat == "INSTRUCTION"    :
            instr_line = item["line"]

        merged_registry[sn] = {
            "category"  : cat,
            "key"       : key,
            "item"      : copy.deepcopy(item),
            "instr_line": instr_line,
        }
        return True

    # Step 1: Add all target items as the base, skipping any in the remove list
    for sn in sorted(target_registry.keys()):
        if sn in remove_serials:
            e = target_registry[sn]
            rpt_removed.append(f"  {sn} [{e['category']}] {str(e['key'])[:80]}")
            continue
        e = target_registry[sn]
        if apply_item(e, sn):
            rpt_base.append(f"  {sn} [{e['category']}] {str(e['key'])[:80]}")

    # Step 2: Add items from Widget 5, OR in auto-full-merge mode add ALL source items
    if _auto_full_merge:
        # Add every item from the source space; apply_item deduplicates automatically
        for sn in sorted(source_registry.keys()):
            e = source_registry[sn]
            if apply_item(e, sn):
                rpt_added.append(f"  {sn} [{e['category']}] [auto-merge] {str(e['key'])[:80]}")
    else:
        # Selective add: only the serial numbers entered in Widget 5
        for sn in sorted(add_serials):
            reg, reg_label = _resolve_registry(sn)
            if reg is None or sn not in reg:
                continue
            e = reg[sn]
            if apply_item(e, sn):
                rpt_added.append(f"  {sn} [{e['category']}] [{reg_label}] {str(e['key'])[:80]}")

    # Step 3: Build the final instruction text block
    #
    # FIX - each instruction always stored as its own list element so it appears
    # on its own line (previously added lines were concatenating into one paragraph).
    #
    # Rules:
    #   - Keep target text_instructions, filtered by Widget 6 removals
    #   - Each instruction (kept or added) is ONE element in the content list
    #   - One empty-string separator between the target block and the added block
    #   - In auto-full-merge mode : add ALL source instruction lines not in target
    #   - In selective mode       : add only instruction lines from Widget 5 serials
    #   - Deduplication by norm_key throughout (target version always wins)

    # Collect norm_keys of instructions to remove from the target content
    _rem_instr_nkeys = {
        norm_key(target_registry[sn]["key"])
        for sn in remove_serials
        if sn in target_registry and target_registry[sn]["category"] == "INSTRUCTION"
    }

    # Deep-copy the raw target text_instructions list (preserves all original API fields)
    _raw_ti_list = copy.deepcopy(get_txt(target_ss))
    _ti_shell    = _raw_ti_list[0] if _raw_ti_list else {"content": []}
    _raw_content = _ti_shell.get("content", [])

    # Normalize content (string or list) into a flat list of individual non-empty lines.
    # This is the core fix: every instruction becomes its own list element.
    def _content_to_lines(content):
        """Return a flat list of stripped non-empty instruction lines."""
        if isinstance(content, str):
            # Single string that may contain newline-separated instructions
            return [ln for ln in content.split("\n") if ln.strip()]
        elif isinstance(content, list):
            result = []
            for entry in content:
                # Each list element might itself contain embedded newlines
                for ln in str(entry).split("\n"):
                    if ln.strip():
                        result.append(ln)
            return result
        return []

    _target_lines = _content_to_lines(_raw_content)

    # Filter out removed lines; build a dedup set of norm_keys already present
    _existing_nkeys = set()
    _kept_lines     = []
    for _ln in _target_lines:
        _nk = norm_key(_ln)
        if _nk in _rem_instr_nkeys:
            continue
        _existing_nkeys.add(_nk)
        _kept_lines.append(_ln)

    # Collect new instruction lines to append (one entry per instruction = one line)
    _lines_added = []

    if _auto_full_merge:
        # Auto-full-merge: add ALL instruction lines from the source space
        for sn in sorted(source_registry.keys()):
            e = source_registry[sn]
            if e["category"] != "INSTRUCTION":
                continue
            _line_text = e["item"]["line"]
            _nk = norm_key(_line_text)
            if _nk not in _existing_nkeys:
                _existing_nkeys.add(_nk)
                _lines_added.append(_line_text)
    else:
        # Selective: add only instruction lines from Widget 5 serial numbers
        for sn in sorted(add_serials):
            reg, _ = _resolve_registry(sn)
            if reg is None or sn not in reg:
                continue
            e = reg[sn]
            if e["category"] != "INSTRUCTION":
                continue
            _line_text = e["item"]["line"]
            _nk = norm_key(_line_text)
            if _nk not in _existing_nkeys:
                _existing_nkeys.add(_nk)
                _lines_added.append(_line_text)

    # Separator: one blank element between the two blocks (only when both have content)
    _separator_lines = [""] if (_kept_lines and _lines_added) else []

    # Final content: each instruction is its own list element -> renders on its own line
    _final_lines = _kept_lines + _separator_lines + _lines_added

    if _final_lines or _raw_ti_list:
        # Always store as a list so every instruction appears on a separate line.
        # The Genie API accepts both str and list for content; list preserves line breaks.
        _ti_shell["content"] = _final_lines
        inst["text_instructions"] = [_ti_shell]

    # Counts for the summary report
    _n_target_kept = len(_kept_lines)
    _n_removed     = len(_rem_instr_nkeys)
    _n_added       = len(_lines_added)

    # Print merge summary
    _merge_mode_label = "AUTO FULL MERGE" if _auto_full_merge else "Selective Merge"
    body = (
        f"Mode    : {_merge_mode_label}\n"
        f"Target  : {target_label}  [{TARGET_SPACE_ID}]\n"
        f"Source  : {source_label}  [{SOURCE_SPACE_ID}]\n\n"
        f"Base items from target kept : {len(rpt_base)}\n"
        f"Items removed from target   : {len(rpt_removed)}\n"
        f"Items added from source     : {len(rpt_added)}\n"
        f"Duplicates suppressed       : {len(rpt_skipped)}"
    )
    if rpt_removed:
        body += "\n\nRemoved from target:\n" + "\n".join(rpt_removed)
    if rpt_added:
        body += "\n\nAdded from source:\n" + "\n".join(rpt_added)
    if rpt_skipped:
        body += "\n\nSkipped (duplicate key, target version kept):\n" + "\n".join(rpt_skipped)

    console.print(Panel(body, title=f"Merged Space Built  [{_merge_mode_label}]", border_style="#005500"))
    console.print(f"  Tables           : {len(ds['tables'])}")
    console.print(f"  Joins            : {len(inst['join_specs'])}")
    _merged_ti   = inst.get("text_instructions", [])
    _total_lines = len([l for l in (_merged_ti[0].get("content", []) if _merged_ti else []) if str(l).strip()])
    console.print(f"  Instructions     : {_n_target_kept} kept + {_n_added} added  ({_n_removed} removed)  = {_total_lines} lines total")
    console.print(f"  Measures         : {len(snip['measures'])}")
    console.print(f"  Expressions      : {len(snip['expressions'])}")
    console.print(f"  Filters          : {len(snip['filters'])}")
    console.print(f"  Benchmarks       : {len(merged_ss['benchmarks']['questions'])}")
    console.print(f"  Sample Questions : {len(merged_ss['config']['sample_questions'])}")
    console.print(f"  Example SQLs     : {len(inst['example_question_sqls'])}")

    sp2_merged_ss = merged_ss
    sp2_filtered  = False

    console.print("\nRun Cell 6 to clean, sort, and validate.")

# CELL 6 - CLEAN + SORT + VALIDATE

mode = dbutils.widgets.get("mode")

if mode == "merge":

    # Fix description fields - normalize to a clean list or remove if empty
    def fix_desc(v):
        if isinstance(v, list):
            c = [str(x) for x in v if x is not None and str(x).strip()]
            return c if c else None
        if isinstance(v, str) and v.strip(): return [v.strip()]
        return None

    # Sanitize table and column description fields
    def sanitize(ss):
        for t in ss.get("data_sources", {}).get("tables", []):
            if not isinstance(t, dict): continue
            if "description" in t:
                f = fix_desc(t["description"])
                if f: t["description"] = f
                else: del t["description"]
            for c in t.get("column_configs", []):
                if not isinstance(c, dict): continue
                if "description" in c:
                    f = fix_desc(c["description"])
                    if f: c["description"] = f
                    else: del c["description"]

    # Sort all sections by their id field (required by the Databricks API)
    def sort_all(ss):
        by_id    = lambda l: sorted(l, key=lambda x: x.get("id", "z" * 32) if isinstance(x, dict) else "z" * 32)
        by_ident = lambda l: sorted(l, key=lambda x: x.get("identifier", "") if isinstance(x, dict) else "")
        ds, inst = ss.get("data_sources", {}), ss.get("instructions", {})
        snip = inst.get("sql_snippets", {})
        bm   = ss.get("benchmarks", {})
        cfg  = ss.get("config", {})
        if "tables"       in ds: ds["tables"]       = by_ident(ds["tables"])
        if "metric_views" in ds: ds["metric_views"]  = by_ident(ds["metric_views"])
        if "join_specs"            in inst: inst["join_specs"]            = by_id(inst["join_specs"])
        if "example_question_sqls" in inst: inst["example_question_sqls"] = by_id(inst["example_question_sqls"])
        if "text_instructions"     in inst: inst["text_instructions"]     = by_id(inst["text_instructions"])
        for k in ("filters", "expressions", "measures"):
            if k in snip: snip[k] = by_id(snip[k])
        if isinstance(bm, dict) and "questions" in bm: bm["questions"] = by_id(bm["questions"])
        if "sample_questions" in cfg: cfg["sample_questions"] = by_id(cfg["sample_questions"])

    ready_ss = copy.deepcopy(merged_ss)
    sanitize(ready_ss)
    sort_all(ready_ss)

    # Validate that the output is valid JSON
    try:
        s = json.dumps(ready_ss, ensure_ascii=False)
        json.loads(s)
        console.print(f"  ready_ss - valid JSON  {len(s):,} chars")
    except Exception as e:
        console.print(f"  ready_ss JSON validation failed: {e}"); raise

    ready_sp2_ss = ready_ss

    _inst  = ready_ss.get("instructions", {})
    _snip  = _inst.get("sql_snippets", {})
    _bm    = ready_ss.get("benchmarks", {})

    def _ids(lst): return [x.get("id", "") for x in lst if isinstance(x, dict)]

    # Check that a list is sorted by id and print the result
    def _chk(lst, label):
        ids = _ids(lst)
        ok  = ids == sorted(ids)
        status = "sorted" if ok else "NOT SORTED - API will reject"
        console.print(f"  {label:<28} : {len(lst):>4} items   {status}")

    txt = len(_inst.get("text_instructions", []))
    console.print(f"  text_instructions blocks : {txt}  (expected 1 or less)")
    _chk(_inst.get("join_specs", []),            "join_specs")
    _chk(_inst.get("example_question_sqls", []), "example_question_sqls")
    _chk(_inst.get("text_instructions", []),     "text_instructions")
    _chk(_snip.get("filters", []),               "filters")
    _chk(_snip.get("expressions", []),           "expressions")
    _chk(_snip.get("measures", []),              "measures")
    _chk(_bm.get("questions", []) if isinstance(_bm, dict) else [], "benchmarks.questions")
    _chk(ready_ss.get("config", {}).get("sample_questions", []),    "sample_questions")
    console.print("\nAll sort checks done. Run Cell 7 to snapshot, export JSON, and generate PDF.")

    console.print("Mode = rollback - Cell 6 skipped.")

#   snapshot_SP1_<ts>.json         - pre-merge snapshot of Space 1
#   snapshot_SP2_<ts>.json         - pre-merge snapshot of Space 2
#   genie_SP1_EXPORT_<ts>.json     - Space 1 serialized export
#   genie_MERGED_PATCH_<ts>.json   - merged space that patches the target
#   genie_comparison_<ts>.pdf      - side-by-side PDF report

mode = dbutils.widgets.get("mode")

if mode == "merge":

    ts = datetime.utcnow().strftime("%Y%m%d_%H%M%S")

    DBFS_SNAPSHOT = "dbfs:/FileStore/genie_snapshots"

    # Write a JSON file to workspace; fall back to DBFS if the workspace write fails
    def write_json(path, data, label):
        try:
            os.makedirs(os.path.dirname(path), exist_ok=True)
            with open(path, "w", encoding="utf-8") as f:
                json.dump(data, f, indent=2, ensure_ascii=False)
            console.print(f"  {label} -> {path}")
        except OSError as e:
            console.print(f"  {label}: workspace write failed ({e}) - using DBFS")
            fb = f"{DBFS_SNAPSHOT}/{os.path.basename(path)}"
            dbutils.fs.put(fb, json.dumps(data, indent=2, ensure_ascii=False), overwrite=True)
            console.print(f"  {label} -> {fb}  (DBFS)")

    # Copy a file to the destination; fall back to DBFS if the workspace copy fails
    def copy_file(src, dst, label):
        try:
            os.makedirs(os.path.dirname(dst), exist_ok=True)
            shutil.copy2(src, dst)
            console.print(f"  {label} -> {dst}")
        except OSError as e:
            console.print(f"  {label}: workspace copy failed ({e}) - using DBFS")
            fb = f"{DBFS_SNAPSHOT}/{os.path.basename(dst)}"
            dbutils.fs.cp(f"file://{src}", fb, recurse=False)
            console.print(f"  {label} -> {fb}  (DBFS)")

    snap1_name  = f"snapshot_{SPACE_ID_1}_{ts}.json"
    snap2_name  = f"snapshot_{SPACE_ID_2}_{ts}.json"
    sp1_export  = f"genie_SP1_EXPORT_{ts}.json"
    patch_name  = f"genie_MERGED_PATCH_{TARGET_SPACE_ID}_{ts}.json"
    pdf_name    = f"genie_comparison_{SPACE_ID_1}_vs_{SPACE_ID_2}_{ts}.pdf"

    write_json(f"{SNAPSHOT_FOLDER}/{snap1_name}", {
        "space_id"        : outer1.get("space_id", ""),
        "title"           : outer1.get("title", ""),
        "description"     : outer1.get("description", ""),
        "warehouse_id"    : outer1.get("warehouse_id", ""),
        "parent_path"     : outer1.get("parent_path", ""),
        "serialized_space": outer1.get("serialized_space", ""),
    }, "Snapshot SP1")

    write_json(f"{SNAPSHOT_FOLDER}/{snap2_name}", {
        "space_id"        : outer2.get("space_id", ""),
        "title"           : outer2.get("title", ""),
        "description"     : outer2.get("description", ""),
        "warehouse_id"    : outer2.get("warehouse_id", ""),
        "parent_path"     : outer2.get("parent_path", ""),
        "serialized_space": outer2.get("serialized_space", ""),
    }, "Snapshot SP2")

    write_json(f"{SNAPSHOT_FOLDER}/{sp1_export}", {
        "space_id"        : SPACE_ID_1,
        "title"           : outer1.get("title", ""),
        "description"     : outer1.get("description", ""),
        "warehouse_id"    : outer1.get("warehouse_id", ""),
        "parent_path"     : outer1.get("parent_path", ""),
        "serialized_space": outer1.get("serialized_space", ""),
    }, "SP1 export JSON")

    write_json(f"{SNAPSHOT_FOLDER}/{patch_name}", {
        "space_id"        : TARGET_SPACE_ID,
        "title"           : outer_target["title"],
        "description"     : outer_target.get("description", ""),
        "warehouse_id"    : outer_target["warehouse_id"],
        "parent_path"     : outer_target.get("parent_path", ""),
        "serialized_space": json.dumps(ready_ss, indent=2, ensure_ascii=False),
    }, "Merged patch JSON - patches target space")

    # Generate PDF comparison report using reportlab
    tmp_pdf = f"/tmp/{pdf_name}"
    try:
        from reportlab.lib           import colors as RC
        from reportlab.lib.pagesizes import A3, landscape
        from reportlab.lib.styles    import getSampleStyleSheet, ParagraphStyle
        from reportlab.lib.units     import cm
        from reportlab.platypus      import (SimpleDocTemplate, Table as RT,
                                             TableStyle, Paragraph,
                                             Spacer, HRFlowable)

        doc = SimpleDocTemplate(tmp_pdf, pagesize=landscape(A3),
                                leftMargin=1.2*cm, rightMargin=1.2*cm,
                                topMargin=1.5*cm, bottomMargin=1.5*cm)
        ss_  = getSampleStyleSheet()
        s_t  = ParagraphStyle("pt",  parent=ss_["Title"],  fontSize=16, spaceAfter=4)
        s_m  = ParagraphStyle("pm",  parent=ss_["Normal"], fontSize=9,  spaceAfter=2)
        s_h  = ParagraphStyle("ph",  parent=ss_["Normal"], fontSize=7,
                              fontName="Helvetica-Bold", textColor=RC.white)
        s_c  = ParagraphStyle("pc",  parent=ss_["Normal"], fontSize=7,
                              leading=9, wordWrap="CJK")
        s_ct = ParagraphStyle("pct", parent=ss_["Normal"], fontSize=8,
                              fontName="Helvetica-Bold", textColor=RC.white)

        def P(t, st=None):
            return Paragraph(
                str(t or "").replace("&", "&amp;").replace("<", "&lt;").replace(">", "&gt;"),
                st or s_c)

        elems = [
            P("Genie Space Comparison Report", s_t), Spacer(1, .2*cm),
            P(f"Generated  : {datetime.utcnow().strftime('%Y-%m-%d %H:%M:%S UTC')}", s_m),
            P(f"Space 1    : {outer1.get('title','?')}  [{SPACE_ID_1}]", s_m),
            P(f"Space 2    : {outer2.get('title','?')}  [{SPACE_ID_2}]", s_m),
            P(f"Target     : {target_label}  [{TARGET_SPACE_ID}]", s_m),
            P(f"Items SP1  : {len(df_sp1)}   Items SP2: {len(df_sp2)}", s_m),
            P("GREEN = SP1 only  |  BLUE = SP2 only  |  WHITE = both spaces  |  AMBER = modified", s_m),
            Spacer(1, .3*cm),
            HRFlowable(width="100%", thickness=1.5, color=RC.HexColor("#1F4E79")),
            Spacer(1, .3*cm),
        ]

        COL_W  = [2.0*cm, 17.5*cm, 2.0*cm, 17.5*cm]
        HDRS   = [f"SP1\n{SPACE_ID_1}", "SP1 Description / SQL / Details",
                  f"SP2\n{SPACE_ID_2}", "SP2 Description / SQL / Details"]
        HDR_BG = [RC.HexColor("#1B5E20"), RC.HexColor("#2E7D32"),
                  RC.HexColor("#01579B"), RC.HexColor("#0277BD")]

        C_SP1A = RC.HexColor("#d4edda"); C_SP1B = RC.HexColor("#b8dfc4")
        C_SP2A = RC.HexColor("#cce5ff"); C_SP2B = RC.HexColor("#a8d1f0")
        C_BOTA = RC.white;              C_BOTB = RC.HexColor("#F5F5F5")
        C_CAT  = RC.HexColor("#37474F")

        td = [[P(h, s_h) for h in HDRS]]
        sc = [
            ("FONTSIZE",      (0, 0), (-1, -1), 7),
            ("LEADING",       (0, 0), (-1, -1), 9),
            ("VALIGN",        (0, 0), (-1, -1), "TOP"),
            ("ALIGN",         (0, 0), (-1, -1), "LEFT"),
            ("GRID",          (0, 0), (-1, -1), .3, RC.HexColor("#B0BEC5")),
            ("TOPPADDING",    (0, 0), (-1, -1), 3),
            ("BOTTOMPADDING", (0, 0), (-1, -1), 3),
            ("LEFTPADDING",   (0, 0), (-1, -1), 3),
            ("RIGHTPADDING",  (0, 0), (-1, -1), 3),
        ]
        for ci, bg in enumerate(HDR_BG):
            sc += [("BACKGROUND", (ci, 0), (ci, 0), bg),
                   ("TEXTCOLOR",  (ci, 0), (ci, 0), RC.white)]

        ri = 1
        for cat in CATEGORIES:
            s1_ = df_sp1[df_sp1["category"] == cat].copy()
            s2_ = df_sp2[df_sp2["category"] == cat].copy()
            m1_, m2_ = {}, {}
            for _, r in s1_.iterrows(): m1_.setdefault(r["norm_key"], r)
            for _, r in s2_.iterrows(): m2_.setdefault(r["norm_key"], r)
            ks_ = sorted(set(m1_) | set(m2_))
            if not ks_: continue

            cat_label = CAT_LABELS.get(cat, cat)
            td.append([P(f"  {cat_label}", s_ct)] + [P("")] * 3)
            sc += [("BACKGROUND", (0, ri), (-1, ri), C_CAT),
                   ("SPAN",       (0, ri), (-1, ri))]
            ri += 1

            for i, k in enumerate(ks_):
                r1_, r2_ = m1_.get(k), m2_.get(k)
                o1 = r1_ is not None and r2_ is None
                o2 = r1_ is None     and r2_ is not None
                bg = (C_SP1A if i % 2 == 0 else C_SP1B) if o1 else \
                     (C_SP2A if i % 2 == 0 else C_SP2B) if o2 else \
                     (C_BOTA if i % 2 == 0 else C_BOTB)
                td.append([
                    P(r1_["serial_no"]    if r1_ is not None else ""),
                    P(r1_["display_desc"] if r1_ is not None else ""),
                    P(r2_["serial_no"]    if r2_ is not None else ""),
                    P(r2_["display_desc"] if r2_ is not None else ""),
                ])
                sc.append(("BACKGROUND", (0, ri), (-1, ri), bg)); ri += 1

        rl_tbl = RT(td, colWidths=COL_W, repeatRows=1)
        rl_tbl.setStyle(TableStyle(sc))
        elems.append(rl_tbl)
        doc.build(elems)
        copy_file(tmp_pdf, f"{SNAPSHOT_FOLDER}/{pdf_name}", "PDF report")

    except ImportError:
        import subprocess
        subprocess.run(["pip", "install", "reportlab", "--quiet"], check=True)
        console.print("  reportlab installed - please re-run Cell 7.")
    except Exception as ex:
        import traceback
        console.print(f"  PDF generation failed: {ex}")
        traceback.print_exc()

    console.print(Panel(
        f"Saved to : {SNAPSHOT_FOLDER}\n"
        f"  Snapshot SP1      : {snap1_name}\n"
        f"  Snapshot SP2      : {snap2_name}\n"
        f"  SP1 export        : {sp1_export}\n"
        f"  Merged patch      : {patch_name}  (patches target space)\n"
        f"  PDF report        : {pdf_name}\n\n"
        f"Target Space  : {target_label}  [{TARGET_SPACE_ID}]\n"
        f"Source Space  : {source_label}  [{SOURCE_SPACE_ID}]\n\n"
        "Run Cell 8 to CREATE a new Genie space (optional).\n"
        "Run Cell 9 to PATCH the target space with merged content.",
        title="Cell 7 complete", border_style="#005500"))

else:
    console.print("Mode = rollback - Cell 7 skipped.")

if mode == "merge":

    dbutils.widgets.text("new_space_title", "",

    new_title = dbutils.widgets.get("new_space_title").strip()

else:
    console.print("Mode = rollback - Cell 8 skipped.")


# CELL 9 - PATCH TARGET SPACE
#
# Patches the target space (selected in Widget 4) with the merged content.
# The source space is not modified.

mode = dbutils.widgets.get("mode")

if mode == "merge":

    console.print(Panel(
        f"Target (will be patched)  : {target_label}  [{TARGET_SPACE_ID}]\n"
        f"Source (unchanged)        : {source_label}  [{SOURCE_SPACE_ID}]\n\n"
        f"Items in merged space:\n"
        f"  Tables           : {len(ready_ss.get('data_sources',{}).get('tables',[]))}\n"
        f"  Joins            : {len(ready_ss.get('instructions',{}).get('join_specs',[]))}\n"
        f"  Measures         : {len(ready_ss.get('instructions',{}).get('sql_snippets',{}).get('measures',[]))}\n"
        f"  Expressions      : {len(ready_ss.get('instructions',{}).get('sql_snippets',{}).get('expressions',[]))}\n"
        f"  Filters          : {len(ready_ss.get('instructions',{}).get('sql_snippets',{}).get('filters',[]))}\n"
        f"  Example SQLs     : {len(ready_ss.get('instructions',{}).get('example_question_sqls',[]))}\n"
        f"  Benchmarks       : {len(ready_ss.get('benchmarks',{}).get('questions',[]))}\n"
        f"  Sample Questions : {len(ready_ss.get('config',{}).get('sample_questions',[]))}",
        title="Cell 9 - Patch Target Space", border_style="yellow"))

    # Fetch the current live version just before patching.
    # Databricks uses optimistic concurrency - version must match exactly.
    # Any re-run or concurrent edit since Cell 2 would cause a 409 conflict without this refresh.
    _ver_r = requests.get(
        f"{DATABRICKS_HOST}/api/2.0/genie/spaces/{TARGET_SPACE_ID}",
        headers=HEADERS,
        params={"include_serialized_space": "true"},
        timeout=30,
    )
    if _ver_r.status_code == 200:
        _ver_raw = _ver_r.json().get("serialized_space", "{}")
        _live_ver = (json.loads(_ver_raw) if isinstance(_ver_raw, str) else _ver_raw).get("version", 1)
        ready_ss["version"] = _live_ver
        console.print(f"  Live target version refreshed: {_live_ver}")
    else:
        console.print(f"  Could not refresh target version (HTTP {_ver_r.status_code}) - proceeding with cached version")

    resp_p = requests.patch(
        f"{DATABRICKS_HOST}/api/2.0/genie/spaces/{TARGET_SPACE_ID}",
        headers=HEADERS,
        json={
            "title"           : outer_target["title"],
            "description"     : outer_target.get("description", ""),
            "warehouse_id"    : outer_target["warehouse_id"],
            "serialized_space": json.dumps(ready_ss, ensure_ascii=False),
        },
        timeout=120,
    )

    if resp_p.status_code == 200:
        console.print(Panel(
            f"Target space patched successfully.\n\n"
            f"  Patched   : '{outer_target['title']}'  [{TARGET_SPACE_ID}]\n"
            f"  Untouched : '{outer_source['title']}'  [{SOURCE_SPACE_ID}]\n\n"
            f"To rollback:\n"
            f"  1. Set mode -> rollback\n"
            f"  2. Enter snapshot -> {snap1_name if is_sp1_target else snap2_name}\n"
            f"  3. Run Cell 10",
            title="Cell 9 - Patch complete", border_style="#005500"))
    else:
        console.print(f"  Error  HTTP {resp_p.status_code}")
        console.print(f"  {resp_p.text[:800]}")
        if resp_p.status_code == 409:
            console.print("  409 = version conflict. The space was modified after the version refresh.")
            console.print("  Re-run from Cell 2 to re-fetch, then retry Cell 9.")
        else:
            console.print("  Check CAN EDIT permission on the target space.")

else:
    console.print("Mode = rollback - Cell 9 skipped.")


# CELL 10 - ROLLBACK

mode = dbutils.widgets.get("mode")

if mode == "rollback":

    rb_file = dbutils.widgets.get("rollback_file").strip()
    if not rb_file:
        console.print("Enter snapshot filename in Widget 2.")
        raise ValueError("rollback_file widget is empty.")

    snap_path = f"{SNAPSHOT_FOLDER}/{rb_file}"
    console.print(f"Loading : {snap_path}")

    try:
        with open(snap_path, "r", encoding="utf-8") as f:
            rb = json.load(f)
        console.print("  Snapshot loaded.")
    except FileNotFoundError:
        console.print(f"  Not found: {snap_path}"); raise
    except Exception as e:
        console.print(f"  Load error: {e}"); raise

    sid  = rb.get("space_id", "")
    raw  = rb.get("serialized_space", "{}")
    ss_r = json.loads(raw) if isinstance(raw, str) else raw
    ds_r = ss_r.get("data_sources", {})
    in_r = ss_r.get("instructions", {})
    sn_r = in_r.get("sql_snippets", {})
    bm_r = ss_r.get("benchmarks", {})
    bmc  = (len(bm_r.get("questions", [])) if isinstance(bm_r, dict)
            else len(bm_r) if isinstance(bm_r, list) else 0)

    console.print(Panel(
        f"Title        : '{rb.get('title','?')}'\nSpace ID     : {sid}\n"
        f"Tables       : {len(ds_r.get('tables',[]))}\n"
        f"Joins        : {len(in_r.get('join_specs',[]))}\n"
        f"Example SQLs : {len(in_r.get('example_question_sqls',[]))}\n"
        f"Measures     : {len(sn_r.get('measures',[]))}\n"
        f"Expressions  : {len(sn_r.get('expressions',[]))}\n"
        f"Filters      : {len(sn_r.get('filters',[]))}\n"
        f"Benchmarks   : {bmc}\n"
        f"Sample Qs    : {len(ss_r.get('config',{}).get('sample_questions',[]))}",
        title="Restoring snapshot", border_style="yellow"))

    # Fetch the current live version before patching.
    # The snapshot version is stale after a successful merge patch, so we inject
    # the current live version into ss_r to avoid a 409 version conflict.
    _rb_ver_r = requests.get(
        f"{DATABRICKS_HOST}/api/2.0/genie/spaces/{sid}",
        headers=HEADERS,
        params={"include_serialized_space": "true"},
        timeout=30,
    )
    if _rb_ver_r.status_code == 200:
        _rb_raw = _rb_ver_r.json().get("serialized_space", "{}")
        _rb_live_ver = (json.loads(_rb_raw) if isinstance(_rb_raw, str) else _rb_raw).get("version", 1)
        ss_r["version"] = _rb_live_ver
        console.print(f"  Live space version refreshed: {_rb_live_ver}  (snapshot version replaced)")
    else:
        console.print(f"  Could not refresh live version (HTTP {_rb_ver_r.status_code}) - rollback may get 409")

    resp_rb = requests.patch(
        f"{DATABRICKS_HOST}/api/2.0/genie/spaces/{sid}",
        headers=HEADERS,
        json={
            "title"           : rb["title"],
            "description"     : rb.get("description", ""),
            "warehouse_id"    : rb["warehouse_id"],
            # Use ss_r (not the raw snapshot string) so the refreshed live version is included
            "serialized_space": json.dumps(ss_r, ensure_ascii=False),
        },
        timeout=120,
    )

    if resp_rb.status_code == 200:
        console.print(Panel(
            f"Rollback successful.\n"
            f"  '{rb['title']}'  [{sid}]  restored to pre-merge state.\n"
            "  Open the space in Databricks to verify.",
            title="Cell 10 - Rollback complete", border_style="#005500"))
    else:
        console.print(f"  HTTP {resp_rb.status_code}")
        console.print(f"  {resp_rb.text[:600]}")
        if resp_rb.status_code == 409:
            console.print("  409 = version conflict - space was modified after the version refresh.")
            console.print("  Wait a moment and re-run Cell 10 to pick up the latest version.")
        else:
            console.print("  Check CAN EDIT permission on the target space.")

else:
    console.print("Mode = merge - Cell 10 (rollback) skipped.")
