from pydub import AudioSegment
from pathlib import Path

# Point pydub to ffmpeg/ffprobe explicitly
AudioSegment.converter = "/usr/bin/ffmpeg"
AudioSegment.ffprobe = "/usr/bin/ffprobe"

# ------------------------------------------------------------
# 👉 UPDATE THESE TWO LINES WITH YOUR ACTUAL FILE PATHS
# ------------------------------------------------------------
input_file = "/Workspace/Users/chandrabhanu.x.chandrabhanu@gsk.com/FFN_Input/Sjukvården-SE-sound.m4a"
output_file = "/Workspace/Users/chandrabhanu.x.chandrabhanu@gsk.com/FFN_Input/Sjukvården-SE-sound.wav"
# ------------------------------------------------------------

def convert_single_file(input_path: str, output_path: str) -> str:
    """Convert one M4A file to WAV."""
    try:
        # Verify input file exists first
        if not Path(input_path).exists():
            return f"❌ File not found: {input_path}"

        audio = AudioSegment.from_file(input_path, format="m4a")
        audio.export(output_path, format="wav")
        return f"✅ Converted: {input_path} -> {output_path}"
    except Exception as e:
        return f"❌ Failed: {input_path} | Error: {e}"

# Run the conversion
result = convert_single_file(input_file, output_file)
print(result)
